39 template <int MAX_BITS>
40 void zstdHuffBitPacker(hls::stream<DSVectorStream_dt<HuffmanCode_dt<MAX_BITS>, 1> >& hfEncodedStream,
41  hls::stream<IntVectorStream_dt<8, 2> >& hfBitStream) {
42  // pack huffman codes from multiple input code streams
43  bool done = false;
44  IntVectorStream_dt<8, 2> outVal;
45  ap_uint<32> outReg = 0;
46  while (!done) {
47  done = true;
48  outVal.strobe = 2;
49  uint8_t validBits = 0;
50  uint16_t outCnt = 0;
51  hf_bitPacking:
52  for (auto inVal = hfEncodedStream.read(); inVal.strobe > 0; inVal = hfEncodedStream.read()) {
53 #pragma HLS PIPELINE II = 1
54  done = false;
55  outReg.range((uint8_t)(inVal.data[0].bitlen) + validBits - 1, validBits) = inVal.data[0].code;
56  validBits += (uint8_t)(inVal.data[0].bitlen);
57  if (validBits > 15) {
58  validBits -= 16;
59  outVal.data[0] = outReg.range(7, 0);
60  outVal.data[1] = outReg.range(15, 8);
61  // printf("%u. huff bitstream: %u\n", outCnt, (uint8_t) outVal.data[0]);
62  // printf("%u. huff bitstream: %u\n", outCnt + 1, (uint8_t) outVal.data[1]);
63  outReg >>= 16;
64  hfBitStream << outVal;
65  outCnt += 2;
66  }
67  }
68  if (outCnt) {
69  // mark end by adding 1-bit "1"
70  outReg |= (uint32_t)1 << validBits;
71  ++validBits;
72  }
73  // write remaining bits
74  if (validBits) {
75  outVal.strobe = (validBits + 7) >> 3;
76  outVal.data[0] = outReg.range(7, 0);
77  outVal.data[1] = outReg.range(15, 8); // dummy if strobe = 1
78  // printf("%u. huff bitstream: %u\n", outCnt, (uint8_t) outVal.data[0]);
79  // if (outVal.strobe > 1) printf("%u. huff bitstream: %u\n", outCnt + 1, (uint8_t) outVal.data[1]);
80  hfBitStream << outVal;
81  outCnt += outVal.strobe;
82  } // exit(0);
83  // printf("bitpacker written bytes: %u\n", outCnt);
84  outVal.strobe = 0;
85  hfBitStream << outVal;
86  }
87 }

89 template <int MAX_BITS, int INSTANCES = 1>
90 void zstdHuffmanEncoder(hls::stream<IntVectorStream_dt<8, 1> >& inValStream,
91  hls::stream<bool>& rleFlagStream,
92  hls::stream<DSVectorStream_dt<HuffmanCode_dt<MAX_BITS>, 1> >& hfCodeStream,
93  hls::stream<DSVectorStream_dt<HuffmanCode_dt<MAX_BITS>, 1> >& hfEncodedStream,
94  hls::stream<ap_uint<16> >& hfLitMetaStream) {
95  // read huffman table
96  HuffmanCode_dt<MAX_BITS> hfcTable[256]; // use LUTs for implementation as registers
97  DSVectorStream_dt<HuffmanCode_dt<MAX_BITS>, 1> outVal;
98  bool done = false;
99 
100  while (true) {
101  uint16_t hIdx = 0;
102  // pre-read value to check strobe
103  auto inVal = inValStream.read();
104  if (inVal.strobe == 0) break;
105  // Check if this literal block is RLE type
106  bool isRle = rleFlagStream.read();
107  // read this table only once
108  read_hfc_table:
109  for (auto hfCode = hfCodeStream.read(); hfCode.strobe > 0; hfCode = hfCodeStream.read()) {
110 #pragma HLS PIPELINE II = 1
111  hfcTable[hIdx++] = hfCode.data[0];
112  }
113  uint8_t streamCnt = inVal.data[0];
114  hfLitMetaStream << ((uint16_t)streamCnt & 0x000F);
115  // Parallel read 8 * INSTANCES bits of input
116  // Parallel encode to INSTACNCES output streams
117  encode_lit_streams:
118  for (uint8_t si = 0; si < streamCnt; ++si) {
119  ap_uint<16> streamSize;
120  uint16_t streamCmpSize = 0;
121  uint8_t cmpBits = 0;
122  // read present stream size
123  inVal = inValStream.read();
124  streamSize.range(7, 0) = inVal.data[0];
125  inVal = inValStream.read();
126  streamSize.range(15, 8) = inVal.data[0];
127  // hfLitMetaStream << streamSize;
128  // Since for RLE type literals, only one stream is present
129  // in that stream, last literal must not be encoded, comes first in reversed stream
130  if (isRle) {
131  --streamSize;
132  inValStream.read();
133  }
134  outVal.strobe = 1;
135  huff_encode:
136  for (uint16_t i = 0; i < (uint16_t)streamSize; ++i) {
137 #pragma HLS PIPELINE II = 1
138  inVal = inValStream.read();
139  outVal.data[0] = hfcTable[inVal.data[0]];
140  hfEncodedStream << outVal;
141  /*if (outVal.data[0].bitlen == 0) {
142  printf("Error: 0 bitlen for char_num: %u at idx: %u is 0\n", (uint8_t)inVal.data[0], i);
143  exit(0);
144  }*/
145  cmpBits += outVal.data[0].bitlen;
146  if (cmpBits > 7) {
147  streamCmpSize += cmpBits >> 3;
148  cmpBits &= 7;
149  }
150  }
151  // cmpBits cannot be greater than 7, 1 extra bit indicating end of stream
152  if (cmpBits + 1) {
153  ++streamCmpSize;
154  }
155  hfLitMetaStream << streamCmpSize;
156  // printf("huf stream size orig: %u, cmp: %u\n", (uint16_t)streamSize, streamCmpSize);
157  // end of sub-stream
158  outVal.strobe = 0;
159  hfEncodedStream << outVal;
160  }
161  }
162  // end of file
163  hfCodeStream.read();
164  outVal.strobe = 0;
165  hfEncodedStream << outVal;
166 }

168 inline uint8_t getOptimalTableLog(const uint8_t maxLog, uint16_t symbolSize, uint16_t currentMaxCode) {
169 #pragma HLS INLINE off
170  auto maxBitsNeeded = bitsUsed31(symbolSize - 1) - 2;
171  ap_uint<10> tableLog = maxLog;
172  // fin table log
173  uint8_t minBitsSrc = bitsUsed31(symbolSize) + 1;
174  uint8_t minBitsSymbols = bitsUsed31(currentMaxCode) + 2;
175  uint8_t minBitsNeeded = minBitsSrc < minBitsSymbols ? minBitsSrc : minBitsSymbols;
176 
177  if (maxBitsNeeded < tableLog) tableLog = maxBitsNeeded; // Accuracy can be reduced
178  if (minBitsNeeded > tableLog) tableLog = minBitsNeeded; // Need a minimum to safely represent all symbol values
179  if (tableLog < c_fseMinTableLog) tableLog = c_fseMinTableLog;
180  if (tableLog > maxLog) tableLog = maxLog; // will be changed to c_fseMaxTableLog
181  return (uint8_t)tableLog;
182 }

184 template <int MAX_FREQ_DWIDTH = 17>
185 void normalizeFreq(
186  ap_uint<MAX_FREQ_DWIDTH>* inFreq,
187  uint16_t symbolSize, // total : maximum value of symbols (max literal value, number of LL/ML/OF code values: seqcnt)
188  uint16_t curMaxCodeVal, // maxSymbolValue : maximum code value (max bitlen value for literal huffman codes, max
189  // LL/ML/OF code value)
190  uint8_t tableLog,
191  int16_t* normTable) {
192 #pragma HLS INLINE off
193  // core module to generate normalized table
194  uint64_t rtbXvStepTable[8];
195  uint8_t scale = 62 - tableLog;
196  ap_uint<64> step = ((ap_uint<64>)1 << 62) / symbolSize; /* <-- one division here ! */
197  uint64_t vStep = (uint64_t)1 << (scale - 20);
198  int stillToDistribute = 1 << tableLog;
199  uint32_t largest = 0;
200  int16_t largestP = 0;
201  uint32_t lowThreshold = (uint32_t)(symbolSize >> tableLog);
202 // printf("NormalizeFreq - tableLog: %u, total: %u, maxSymbolValue: %u\n", tableLog, symbolSize, curMaxCodeVal);
203 init_rtbTableSteps:
204  for (int i = 0; i < 8; ++i) {
205 #pragma HLS PIPELINE off
206  rtbXvStepTable[i] = vStep * c_rtbTable[i];
207  }
208 
209 norm_count:
210  for (uint16_t s = 0; s <= curMaxCodeVal; ++s) {
211 #pragma HLS PIPELINE II = 1
212  // if (inFreq[s] == symbolSize) return 0; /* rle special case */
213  auto freq = inFreq[s];
214  int16_t proba = (int16_t)((freq * step) >> scale);
215  ap_uint<1> incrProba = (((freq * step) - ((uint64_t)proba << scale)) > rtbXvStepTable[proba]);
216  int16_t val = 0;
217  int16_t valDecrement = 0;
218  if (freq == 0) {
219  val = 0;
220  valDecrement = 0;
221  } else if (freq <= lowThreshold) {
222  val = -1;
223  valDecrement = 1;
224  } else {
225  if (proba < 8) {
226  // uint64_t restToBeat = vStep * c_rtbTable[proba];
227  proba += incrProba;
228  }
229  if (proba > largestP) {
230  largestP = proba;
231  largest = s;
232  }
233  val = proba;
234  valDecrement = proba;
235  }
236  normTable[s] = val;
237  stillToDistribute -= valDecrement;
238  // printf("table: %d rem: %d\n",normTable[s],stillToDistribute);
239  }
240  assert(-stillToDistribute < (normTable[largest] >> 1));
241  // corner case, need another normalization method
242  // FSE_normalizeM2(normTable, tableLog, inFreq, symbolSize, curMaxCodeVal);
243  // printf("Warning: FSE Normalized Table corner case hit!!!\n");
244  normTable[largest] += (short)stillToDistribute;
245  // print normalized table
246  // for (int i = 0; i <= curMaxCodeVal; ++i) printf("%d. normCount: %d\n", i, normTable[i]);
247 }

249 template <int MAX_FREQ_DWIDTH = 17>
250 void normalizedTableGen(hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& inFreqStream,
251  hls::stream<IntVectorStream_dt<16, 1> > normTableStream[2]) {
252  // generate normalized counter to be used for fse table generation
253  /*
254  * > Read the frequencies
255  * > Get max frequency value, maxCode and sequence count
256  * > Calculate optimal table log
257  * > Calculate normalized distribution table
258  */
259  const uint8_t c_maxTableLog[4] = {c_fseMaxTableLogLL, c_fseMaxTableLogOF, c_fseMaxTableLogHF, c_fseMaxTableLogML};
260  const uint8_t c_hfIdx = 2;
261  IntVectorStream_dt<16, 1> outVal;
262  ap_uint<MAX_FREQ_DWIDTH> inFreq[64]; // using max possible size for array
263  int16_t normTable[64];
264  uint16_t seqCnt = 0;
265 norm_tbl_outer_loop:
266  while (true) {
267  uint8_t lastSeq[3] = {0, 0, 0}; // ll, ml, of
268  uint8_t lsk = 0;
269  // first value is sequence count
270  auto inVal = inFreqStream.read();
271  if (inVal.strobe == 0) break;
272  seqCnt = inVal.data[0];
273  // last sequence
274  read_last_seq:
275  for (uint8_t i = 0; i < 3; ++i) {
276 #pragma HLS PIPELINE II = 1
277  inVal = inFreqStream.read();
278  lastSeq[i] = inVal.data[0];
279  }
280  calc_norm_outer:
281  for (uint8_t k = 0; k < 4; ++k) {
282  init_norm_table:
283  for (uint8_t i = 0; i < 64; ++i) {
284 #pragma HLS PIPELINE off
285  normTable[i] = 0;
286  }
287  uint16_t maxSymbol;
288  ap_uint<MAX_FREQ_DWIDTH> maxFreq = 0;
289  uint16_t symCnt = seqCnt;
290  volatile uint16_t symCnt_1 = seqCnt;
291  // read max literal value, before weight freq data
292  if (c_hfIdx == k) {
293  inVal = inFreqStream.read();
294  symCnt = inVal.data[0];
295  symCnt_1 = symCnt;
296  }
297  // number of frequencies to read
298  inVal = inFreqStream.read();
299  uint16_t freqCnt = inVal.data[0];
300  // read input frequencies
301  read_in_freq:
302  for (uint16_t i = 0; i < freqCnt; ++i) {
303 #pragma HLS PIPELINE II = 1
304  inVal = inFreqStream.read();
305  inFreq[i] = inVal.data[0];
306  if (inVal.data[0] > 0) maxSymbol = i;
307  if (inVal.data[0] > maxFreq) maxFreq = inVal.data[0];
308  // if (c_hfIdx == k) printf("weight: %u, freq: %u\n", i, (uint8_t)inVal.data[0]);
309  }
310  // get optimal table log
311  uint8_t tableLog = getOptimalTableLog(c_maxTableLog[k], symCnt, maxSymbol);
312  if (c_hfIdx != k) {
313  if (inFreq[lastSeq[lsk]] > 1) {
314  inFreq[lastSeq[lsk]]--;
315  symCnt_1 -= 1;
316  // printf("%d. decIndx: %u\n", lsk, (uint16_t)lastSeq[lsk]);
317  }
318  ++lsk;
319  }
320  // generate normalized distribution table
321  normalizeFreq<MAX_FREQ_DWIDTH>(inFreq, symCnt_1, maxSymbol, tableLog, normTable);
322  // write normalized table to output
323  outVal.strobe = 1;
324  // write tableLog, max val at the end of table log
325  normTable[63] = maxSymbol;
326  normTable[62] = tableLog;
327  normTable[61] = (int16_t)(c_hfIdx == k); // To be used later for optimization
328  write_norm_table:
329  for (uint16_t i = 0; i < 64; ++i) {
330 #pragma HLS PIPELINE II = 1
331  outVal.data[0] = normTable[i];
332  normTableStream[0] << outVal;
333  normTableStream[1] << outVal;
334  }
335  }
336  }
337  outVal.strobe = 0;
338  normTableStream[0] << outVal;
339  normTableStream[1] << outVal;
340 }

342 void fseHeaderGen(hls::stream<IntVectorStream_dt<16, 1> >& normTableStream,
343  hls::stream<IntVectorStream_dt<8, 2> >& fseHeaderStream) {
344  // generate normalized counter to be used for fse table generation
345  int16_t normTable[64];
346  IntVectorStream_dt<8, 2> outVal;
347 
348 fse_header_gen_outer:
349  while (true) {
350  auto inVal = normTableStream.read();
351  if (inVal.strobe == 0) break;
352  normTable[0] = inVal.data[0];
353  // printf("%u. normTbl: %d\n", 0, normTable[0]);
354  read_norm_table:
355  for (uint8_t i = 1; i < 64; ++i) {
356 #pragma HLS PIPELINE II = 1
357  inVal = normTableStream.read();
358  normTable[i] = inVal.data[0];
359  // printf("%u. normTbl: %d\n", i, normTable[i]);
360  }
361  uint16_t maxCharSize = normTable[63] + 1;
362  uint8_t tableLog = normTable[62];
363  // printf("tableLog: %u\n\n", tableLog);
364 
365  uint16_t tableSize = 1 << tableLog;
366  ap_uint<32> bitStream = 0;
367  int8_t bitCount = 0;
368  uint16_t symbol = 0;
369  uint8_t outCount = 0;
370 
371  /* Table Size */
372  bitStream += (tableLog - c_fseMinTableLog);
373  bitCount += 4;
374 
375  /* Init */
376  int remaining = tableSize + 1; /* +1 for extra accuracy */
377  uint16_t threshold = tableSize;
378  uint8_t nbBits = tableLog + 1;
379  outVal.strobe = 2;
380  uint16_t start = symbol;
381  enum ReadNCountStates { PREV_IS_ZERO, REM_LT_THR, NON_ZERO_CNT };
382  ReadNCountStates fsmState = NON_ZERO_CNT;
383  ReadNCountStates fsmNextState = NON_ZERO_CNT;
384  bool previousIs0 = false;
385  bool skipZeroFreq = true;
386  // printf("FSE Encoded header, maxCharSize: %u, remaining: %d\n", maxCharSize, remaining);
387  gen_fse_header_bitstream:
388  while ((symbol < maxCharSize) && (remaining > 1)) {
389 #pragma HLS PIPELINE II = 1
390  if (fsmState == PREV_IS_ZERO) {
391  if (skipZeroFreq) {
392  if (symbol < maxCharSize && !normTable[symbol]) {
393  ++symbol;
394  // printf("Here 0\n");
395  } else {
396  skipZeroFreq = false;
397  }
398  } else {
399  if (symbol >= start + 24) {
400  start += 24;
401  bitStream += 0xFFFF << bitCount;
402  bitCount += 16;
403  // printf("Here 1\n");
404  } else if (symbol >= start + 3) {
405  start += 3;
406  bitStream += 3 << bitCount;
407  bitCount += 2;
408  // printf("Here 2\n");
409  } else {
410  fsmState = NON_ZERO_CNT;
411  bitStream += (uint32_t)(symbol - start) << bitCount;
412  bitCount += 2;
413  // printf("Here 3\n");
414  }
415  }
416  } else if (fsmState == REM_LT_THR) {
417  --nbBits;
418  threshold >>= 1;
419  // printf("Here 5\n");
420  if (remaining > threshold - 1) {
421  fsmState = fsmNextState;
422  }
423  } else {
424  int16_t count = normTable[symbol++];
425  int max = (2 * threshold) - (1 + remaining);
426  remaining -= (count < 0) ? -count : count;
427  ++count;
428  if (count >= threshold) count += max;
429  bitStream += count << bitCount;
430  bitCount += nbBits;
431  // printf("Here 4, remaining: %d, threshold: %d, nbBits: %d\n", remaining, threshold, nbBits);
432  bitCount -= (uint8_t)(count < max);
433  previousIs0 = (count == 1);
434  start = symbol; // set starting symbol for PREV_IS_ZERO state
435  skipZeroFreq = true; // enable skipping of zero norm values for PREV_IS_ZERO state
436  fsmNextState = (previousIs0 ? PREV_IS_ZERO : NON_ZERO_CNT);
437  fsmState = ((remaining < threshold) ? REM_LT_THR : fsmNextState);
438  }
439  // write output bitstream 16-bits at a time
440  if (bitCount >= 16) {
441  outVal.data[0] = bitStream.range(7, 0);
442  outVal.data[1] = bitStream.range(15, 8);
443  // printf("%u. bitstream: %u\n", outCount, (uint8_t)bitStream);
444  // printf("%u. bitstream: %u\n", outCount + 1, (uint8_t)(bitStream >> 8));
445  fseHeaderStream << outVal;
446  bitStream >>= 16;
447  bitCount -= 16;
448  outCount += 2;
449  }
450  }
451  if (bitCount) {
452  outVal.strobe = (uint8_t)((bitCount + 7) / 8);
453  outVal.data[0] = bitStream.range(7, 0);
454  outVal.data[1] = bitStream.range(15, 8);
455  // printf("%u. bitstream: %u\n", outCount, (uint8_t)bitStream);
456  // if(outVal.strobe > 1) printf("%u. bitstream: %u\n", outCount + 1, (uint8_t)bitStream.range(15, 8));
457  fseHeaderStream << outVal;
458  outCount += outVal.strobe;
459  }
460  // printf("out count: %u\n", outCount);
461  outVal.strobe = 0;
462  fseHeaderStream << outVal;
463  }
464  outVal.strobe = 0;
465  fseHeaderStream << outVal;
466 }

468 void fseEncodingTableGen(hls::stream<IntVectorStream_dt<16, 1> >& normTableStream,
469  hls::stream<IntVectorStream_dt<36, 1> >& fseTableStream) {

523  }
524  }

529  for (uint16_t u = 0; u < tableSize; ++u) {
530 #pragma HLS PIPELINE II = 1
531  auto s = symTable[u];
532  tableU16[intm[s]++] = tableSize + u;
533  }
534  /*if (c_hfIdx != cIdx) {
535  printf("Next state table\n");
536  for (int i = 0; i < tableSize; ++i) printf("%d. nxtState: %u\n", i, tableU16[i]);
537  }*/
538  outVal.strobe = 1;
539  // send tableLog and maxSymbol
540  outVal.data[0].range(7, 0) = tableLog;
541  outVal.data[0].range(35, 8) = maxSymbol;
542  fseTableStream << outVal;

544  // send state table, tableSize on reader side can be calculated using tableLog
545  send_state_table:

547 #pragma HLS PIPELINE II = 1
548  outVal.data[0] = tableU16[i];

553  uint16_t total = 0;

559  uint32_t nBits = 0;
560  int16_t findState = 0;

562  nBits = ((tableLog + 1) << 16) - (1 << tableLog);
563  } else if (nv == 1 || nv == -1) {

565  findState = total - 1;
566  ++total;
567  } else {
568  uint8_t maxBitsOut = tableLog - bitsUsed31(nv - 1);

570  nBits = (maxBitsOut << 16) - minStatePlus;

572  total += nv;
573  }

575  outVal.data[0].range(35, 20) = findState;
576  fseTableStream << outVal;

578  }
579  outVal.strobe = 0;
580  fseTableStream << outVal;
581  cIdx++;
582  if (cIdx == 4) cIdx = 0;
583  }
584  outVal.strobe = 0;
585  fseTableStream << outVal;

667 template <int BITSTREAM_DWIDTH = 64>
668 inline void fseEncodeSymbol(uint8_t symbol,
669  ap_uint<36>* fseStateBitsTable,
670  uint16_t* fseNextStateTable,
671  uint16_t& fseState,
672  ap_uint<BITSTREAM_DWIDTH>& bitstream,
673  uint8_t& bitCount,
674  bool isInit) {
675  // encode a symbol using fse table
676  constexpr uint32_t c_oneBy15Lsh = ((uint32_t)1 << 15);
677  ap_uint<36> stateBitVal = fseStateBitsTable[symbol];
678  uint32_t deltaBits = (uint32_t)(stateBitVal.range(19, 0));
679  int16_t findState = (int16_t)(stateBitVal.range(35, 20));
680  uint32_t nbBits;
681  uint16_t stVal;
682  // printf("symbol: %u, deltaBits: %u\n", (uint16_t)symbol, deltaBits);
683  if (isInit) {
684  nbBits = (deltaBits + c_oneBy15Lsh) >> 16;
685  stVal = (nbBits << 16) - deltaBits;
686  // printf("Init ");
687  } else {
688  nbBits = (uint32_t)(deltaBits + fseState) >> 16;
689  stVal = fseState;
690  // write bits to bitstream
691  bitstream |= ((ap_uint<BITSTREAM_DWIDTH>)(stVal & c_bitMask[nbBits]) << bitCount);
692  bitCount += nbBits;
693  // printf("Next ");
694  }
695  // store current state
696  uint32_t nxIdx = (stVal >> nbBits) + findState;
697  // printf("state: %u, stVal: %u, nBits: %u, findState: %d", nxIdx, stVal, nbBits, findState);
698  // if (nxIdx > 512) {printf("\nError: Invalid state index!!\n"); exit(0);}
699  fseState = fseNextStateTable[nxIdx];
700  // printf(", newState: %d, bCnt: %u\n", fseState, bitCount);
701 }

703 void fseEncodeLitHeader(hls::stream<IntVectorStream_dt<4, 1> >& hufWeightStream,
704  hls::stream<IntVectorStream_dt<36, 1> >& fseLitTableStream,
705  hls::stream<IntVectorStream_dt<8, 2> >& encodedOutStream) {
706  // fse encoding of huffman header for encoded literals
707  constexpr uint32_t c_oneBy15Lsh = ((uint32_t)1 << 15);
708  IntVectorStream_dt<8, 2> outVal;
709  ap_uint<36> fseStateBitsTable[256];
710  uint16_t fseNextStateTable[256];
711  ap_uint<4> hufWeights[256];
712  int blk_n = 0;
713 fse_lit_encode_outer:
714  while (true) {
715  blk_n++;
716  uint8_t tableLog;
717  uint16_t maxSymbol;
718  uint16_t maxFreq;
719  // read fse table
720  uint16_t fIdx = 0;
721  // read FSE encoding tables
722  bool done = readFseTable(fseLitTableStream, fseStateBitsTable, fseNextStateTable, tableLog, maxFreq);
723  if (done) break;
724  // read details::c_maxLitV + 1 data from weight stream
725  read_hf_weights:
726  for (auto inWeight = hufWeightStream.read(); inWeight.strobe > 0; inWeight = hufWeightStream.read()) {
727 #pragma HLS PIPELINE II = 1
728  hufWeights[fIdx] = inWeight.data[0];
729  if (inWeight.data[0] > 0) maxSymbol = fIdx;
730  ++fIdx;
731  }
732  uint16_t preStateVal[2];
733  bool isInit[2] = {true, true};
734  bool stateIdx = 0; // 0 for even, 1 for odd
735  uint8_t bitCount = 0;
736  ap_uint<32> bitstream = 0;
737  outVal.strobe = 2;
738  int outCnt = 0;
739  // encode weights stored in reverse order
740  stateIdx = maxSymbol & 1;
741  // printf("Literal huffman header FSE encoding bitstream, maxSymbol: %u\n", maxSymbol);
742  fse_lit_encode:
743  for (int16_t w = maxSymbol - 1; w > -1; --w) {
744 #pragma HLS PIPELINE II = 1
745  uint8_t symbol = hufWeights[w];
746  fseEncodeSymbol<32>(symbol, fseStateBitsTable, fseNextStateTable, preStateVal[stateIdx], bitstream,
747  bitCount, isInit[stateIdx]);
748  isInit[stateIdx] = false;
749  // write bitstream to output
750  if (bitCount > 15) {
751  // write to output stream
752  outVal.data[0] = bitstream.range(7, 0);
753  outVal.data[1] = bitstream.range(15, 8);
754  encodedOutStream << outVal;
755  // printf("%d. bitstream: %u\n", outCnt, (uint8_t)outVal.data[0]);
756  // printf("%d. bitstream: %u\n", outCnt+1, (uint8_t)outVal.data[1]);
757  bitstream >>= 16;
758  bitCount -= 16;
759  outCnt += 2;
760  }
761  // switch state flow
762  stateIdx = (stateIdx + 1) & 1; // 0 if 1, 1 if 0
763  }
764  // encode last two
765  bitstream |= ((preStateVal[0] & c_bitMask[tableLog]) << bitCount);
766  bitCount += tableLog;
767  bitstream |= ((preStateVal[1] & c_bitMask[tableLog]) << bitCount);
768  bitCount += tableLog;
769  // mark end by adding 1-bit "1"
770  bitstream |= (uint32_t)1 << bitCount;
771  ++bitCount;
772  // max remaining biCount can be 15 + (2 * 6) + 1= 28 bits => 4 bytes
773  int8_t remBytes = (int8_t)((bitCount + 7) / 8);
774  // write bitstream to output
775  write_rem_bytes:
776  while (remBytes > 0) {
777 #pragma HLS PIPELINE II = 1
778  // write to output stream
779  outVal.data[0] = bitstream.range(7, 0);
780  outVal.data[1] = bitstream.range(15, 8);
781  outVal.strobe = ((remBytes > 1) ? 2 : 1);
782  encodedOutStream << outVal;
783  bitstream >>= 16;
784  remBytes -= 2;
785  // printf("%d. bitstream r: %u\n", outCnt, (uint8_t)outVal.data[0]);
786  // if(outVal.strobe > 1) printf("%d. bitstream r: %u\n", outCnt+1, (uint8_t)outVal.data[1]);
787  outCnt += outVal.strobe;
788  }
789  // printf("%d. lit FSE encoded bs size: %u\n", blk_n, outCnt);
790  outVal.strobe = 0;
791  encodedOutStream << outVal;
792  }
793  // dump strobe 0 data
794  hufWeightStream.read();
795  // write end of block
796  outVal.strobe = 0;
797  encodedOutStream << outVal;
798 }
