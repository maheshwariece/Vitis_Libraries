40 template <int PARALLEL_BYTE, int LMO_WIDTH>
41 void alignLiterals(hls::stream<bool>& litlenValidStream,
42  hls::stream<bool>& newBlockFlagStream,
43  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& inLitStream,
44  hls::stream<ap_uint<LMO_WIDTH> >& inLitLenStream,
45  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& litStream,
46  hls::stream<ap_uint<LMO_WIDTH> >& litLenStream) {
47  // align literals to PARALLEL_BYTE boundary as per literal length values
48  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
49  const uint16_t c_accRegWidth = c_streamWidth * 2;
50 align_block_lit_output:
51  while (newBlockFlagStream.read()) {
52  ap_uint<c_accRegWidth> litbuf = 0;
53  uint8_t bytesInAcc = 0;
54  uint8_t bitsInAcc = 0;
55  align_literals:
56  while (litlenValidStream.read()) {
57  ap_uint<LMO_WIDTH> litLen = inLitLenStream.read();
58  litLenStream << litLen;
59  if (litLen > 0) {
60  // if present, write literals aligned to PARALLEL_BYTE
61  uint8_t bytesWritten = 0;
62  uint8_t updBInAcc = bytesInAcc;
63  uint8_t bitsInAcc = bytesInAcc * 8;
64  aligned_literal_write:
65  for (int i = 0; i < litLen; i += PARALLEL_BYTE) {
66 #pragma HLS PIPELINE II = 1
67  if (i < (const int)(litLen - bytesInAcc)) {
68  litbuf.range(bitsInAcc + c_streamWidth - 1, bitsInAcc) = inLitStream.read();
69  updBInAcc += PARALLEL_BYTE;
70  }
71  litStream << litbuf.range(c_streamWidth - 1, 0);
72 
73  if (i > (const int)(litLen - PARALLEL_BYTE)) {
74  bytesWritten = litLen - i;
75  break;
76  }
77  litbuf >>= c_streamWidth;
78  updBInAcc -= PARALLEL_BYTE;
79  }
80  litbuf >>= (bytesWritten * 8);
81  bytesInAcc = updBInAcc - bytesWritten;
82  }
83  }
84  }
85  litLenStream << 0;
86 }

88 template <int PARALLEL_BYTE, int BLOCK_SIZE_KB, int LMO_WIDTH>
89 void decodeSequence(hls::stream<bool>& seqValidStream,
90  hls::stream<uint32_t>& seqMetaStream,
91  hls::stream<uint32_t>& fseTableStream,
92  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& seqDecodeInStream,
93  hls::stream<ap_uint<LMO_WIDTH> >& litLenStream,
94  hls::stream<ap_uint<LMO_WIDTH> >& offsetStream,
95  hls::stream<ap_uint<LMO_WIDTH> >& matLenStream,
96  hls::stream<bool>& litlenValidStream,
97  hls::stream<bool>& newBlockFlagStream) {
98  // decode sequences and output literal lengths, offsets and match lengths
99  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
100  const uint16_t c_accRegWidth = c_streamWidth * 2;
101  uint32_t fseTableLL[512];
102 #pragma HLS BIND_STORAGE variable = fseTableLL type = ram_t2p impl = bram
103  uint32_t fseTableOF[256];
104 #pragma HLS BIND_STORAGE variable = fseTableOF type = ram_t2p impl = bram
105  uint32_t fseTableML[512];
106 #pragma HLS BIND_STORAGE variable = fseTableML type = ram_t2p impl = bram
107  ap_uint<LMO_WIDTH> prevOffsets[3] = {1, 4, 8}; // list of previous 3 offsets
108 #pragma HLS ARRAY_PARTITION variable = prevOffsets complete
109 
110 decodeSequence_main_loop:
111  while (seqValidStream.read()) {
112  uint32_t seqMeta = seqMetaStream.read();
113  xfBlockType_t blockType = (xfBlockType_t)(seqMeta & 0x000000FF);
114  uint32_t blockSize = seqMeta >> 8;
115  newBlockFlagStream << 1;
116  if (blockType == RLE_BLOCK) {
117  litlenValidStream << 1;
118  litLenStream << 1;
119  offsetStream << 0;
120  matLenStream << blockSize - 1;
121  } else if (blockType == RAW_BLOCK) {
122  litlenValidStream << 1;
123  litLenStream << blockSize;
124  offsetStream << 0;
125  matLenStream << 0;
126  } else {
127  // write sequence meta data
128  // Word 0 <blockMeta> ***already decoded
129  // Word 1 <decode_flag> 1 if further decoding needed, else 0
130  // Word 2 <literalCount>
131  // Word 3 <symbolCompressionMode><remBlockSize>
132  // 1 byte 3 bytes
133  // Word 4 <seqCnt>
134  // Word 5 <AccuracyLogs> --> <litlen><offset><matlen> 1 byte each lower-higher
135  // decode fse bitstream
136  uint8_t decode_flag = seqMetaStream.read();
137  uint32_t litCount = seqMetaStream.read();
138  if (decode_flag) {
139  // read all fse tables
140  seqd_read_fse_tables_outer:
141  for (int fti = 0; fti < 3; ++fti) {
142  uint16_t fseVCnt = fseTableStream.read();
143  if (fseVCnt != 0xFFFFFFFF) {
144  seqd_read_fse_tables:
145  for (uint16_t i = 0; i < fseVCnt; ++i) {
146 #pragma HLS PIPELINE II = 1
147  auto tmpv = fseTableStream.read();
148  if (fti == 0) {
149  fseTableLL[i] = tmpv;
150  } else if (fti == 1) {
151  fseTableOF[i] = tmpv;
152  } else {
153  fseTableML[i] = tmpv;
154  }
155  }
156  }
157  }
158 
159  uint32_t smbuf = seqMetaStream.read();
160  uint32_t remBlockSize = smbuf >> 8;
161  uint32_t seqCnt = seqMetaStream.read();
162  uint32_t aclbuf = seqMetaStream.read(); // accuracy logs
163  uint8_t accuracyLog[3] = {(uint8_t)aclbuf, (uint8_t)(aclbuf >> 8), (uint8_t)(aclbuf >> 16)};
164  // Decode the FSE compressed bitstream
165  ap_uint<64> acw = 0;
166  fseDecode<PARALLEL_BYTE, BLOCK_SIZE_KB, LMO_WIDTH>(
167  acw, 0, seqDecodeInStream, fseTableLL, fseTableOF, fseTableML, seqCnt, litCount, remBlockSize,
168  accuracyLog, prevOffsets, litLenStream, offsetStream, matLenStream, litlenValidStream);
169  } else {
170  // zero sequence count condition
171  litlenValidStream << 1;
172  litLenStream << litCount;
173  offsetStream << 0;
174  matLenStream << 0;
175  // update previous offsets
176  prevOffsets[2] = prevOffsets[1];
177  prevOffsets[1] = prevOffsets[0];
178  prevOffsets[0] = 0;
179  }
180  }
181  // clear LZ literal buffer
182  litlenValidStream << 0;
183  }
184  // end of file
185  newBlockFlagStream << 0;
186  matLenStream << 0;
187  offsetStream << 0;
188 }

190 template <int PARALLEL_BYTE, int BLOCK_SIZE_KB>
191 void decodeLiterals(hls::stream<bool>& litValidStream,
192  hls::stream<uint32_t>& litMetaStream,
193  hls::stream<uint32_t>& fseTableStream,
194  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& litDecodeInStream,
195  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& literalStream) {
196  // decode literals and forward to literal stream
197  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
198  const uint16_t c_accRegWidth = c_streamWidth * 2;
199  uint8_t hufWeights[256];
200  uint32_t litFseTable[256];
201 
202  uint8_t huffDecoderTableLog = 6;
203  uint16_t wCnt = 0;
204 
205 decodeLiterals_main_loop:
206  while (litValidStream.read()) {
207  // Literal metadata format
208  // Word 0 is block metadata Word 0
209  uint32_t litMeta = litMetaStream.read();
210  xfBlockType_t blockType = (xfBlockType_t)(litMeta & 0x000000FF);
211  uint32_t blockSize = litMeta >> 8;
212  // for raw and RLE Blocks forward as it is
213  uint32_t lbtWrite = blockSize;
214  if (blockType == RLE_BLOCK) lbtWrite = 1;
215 
216  if (blockType == RLE_BLOCK || blockType == RAW_BLOCK) {
217  // write complete block data as it is
218  decodelit_write_rawrle_block_data:
219  for (uint32_t i = 0; i < lbtWrite; i += PARALLEL_BYTE) {
220 #pragma HLS PIPELINE II = 1
221  ap_uint<c_streamWidth> lit = litDecodeInStream.read();
222  literalStream << lit;
223  }
224  } else {
225  bool quadStream = false;
226  xfLitBlockType_t litBlockType;
227  uint32_t regeneratedSize;
228  uint32_t compressedSize = 0;
229 
230  // literal metadata
231  // Word 1 <litBlockType><quadStream><regeneratedSize>
232  // 7-bits 1-bit 3 bytes
233  litMeta = litMetaStream.read();
234  litBlockType = (xfLitBlockType_t)((uint8_t)litMeta & 3);
235  if (litMeta & 0x00000080) quadStream = true;
236  regeneratedSize = litMeta >> 8;
237  // write literals
238  uint32_t litSize2Write = regeneratedSize;
239  if (litBlockType == RLE_LBLOCK) {
240  // pass the RLE literal
241  ap_uint<c_streamWidth> tbuf = litDecodeInStream.read();
242  uint8_t rleLit = tbuf;
243  decodelit_prep_rlelit_buff:
244  for (uint8_t i = 1; i < PARALLEL_BYTE; ++i) {
245 #pragma HLS UNROLL
246  tbuf.range(((i + 1) * 8) - 1, i * 8) = rleLit;
247  }
248  decodelit_write_rlelit_data:
249  for (uint32_t i = 0; i < litSize2Write; i += PARALLEL_BYTE) {
250 #pragma HLS PIPELINE II = 1
251  literalStream << tbuf;
252  }
253  } else if (litBlockType == RAW_LBLOCK) {
254  // pass the raw block
255  decodelit_write_rawlit_data:
256  for (int i = 0; i < litSize2Write; i += PARALLEL_BYTE) {
257 #pragma HLS PIPELINE II = 1
258  ap_uint<c_streamWidth> lit = litDecodeInStream.read();
259  literalStream << lit;
260  }
261  } else {
262  // Literal metadata
263  // Word 2 <huffmanHeader><compressedSize>
264  // 1 byte 3 bytes
265  litMeta = litMetaStream.read();
266  uint8_t hufHeader = (uint8_t)litMeta;
267  uint8_t remBytesInFse = 0;
268  compressedSize = litMeta >> 8;
269  litSize2Write = compressedSize;
270 
271  ap_uint<c_accRegWidth> accHuff = 0;
272  uint8_t bytesInAcc = 0;
273  uint32_t remBytes = 0;
274  if (litBlockType == CMP_LBLOCK) {
275  uint8_t accuracyLog = 6;
276  uint16_t fwCnt = 0;
277  // init huffman weights
278  for (int i = 0; i < 256; i += 2) {
279  hufWeights[i] = 0;
280  hufWeights[i + 1] = 0;
281  }
282  if (hufHeader < 128) {
283  // read FSE table
284  uint32_t tblVCnt = fseTableStream.read();
285  lit_read_fse_table:
286  for (int i = 0; i < tblVCnt; ++i) {
287 #pragma HLS PIPELINE II = 1
288  litFseTable[i] = fseTableStream.read();
289  }
290  // Word 3 <accuracyLog><byteUsedByFSE>
291  // 1 byte 1 byte
292  litMeta = litMetaStream.read();
293  accuracyLog = (uint8_t)litMeta;
294  remBytesInFse = hufHeader - (litMeta >> 8);
295  // decode fse bitstream of huffman weights
296  fseDecodeHuffWeight<PARALLEL_BYTE>(litDecodeInStream, remBytesInFse, accHuff, bytesInAcc,
297  accuracyLog, litFseTable, hufWeights, fwCnt,
298  huffDecoderTableLog);
299  remBytes = compressedSize - hufHeader - 1;
300  wCnt = fwCnt;
301  } else {
302  // 4-bit huffman weights
303  uint16_t hwCnt = hufHeader - 127;
304  uint32_t totalWeights = 0;
305  if (bytesInAcc == 0) {
306  accHuff.range(c_streamWidth - 1, 0) = litDecodeInStream.read();
307  bytesInAcc = PARALLEL_BYTE;
308  }
309  huffman_decode_weights:
310  for (uint8_t i = 0; i < hwCnt; i += 2) {
311 #pragma HLS PIPELINE II = 1
312  if (bytesInAcc == 0) {
313  accHuff.range(c_streamWidth - 1, 0) = litDecodeInStream.read();
314  bytesInAcc = PARALLEL_BYTE;
315  }
316  uint8_t w8t = accHuff;
317  accHuff >>= 8;
318  --bytesInAcc;
319  // get huffman weights
320  uint8_t hfw0 = (w8t >> 4);
321  uint8_t hfw1 = (w8t & 0x0F);
322  hufWeights[i] = hfw0;
323  hufWeights[i + 1] = hfw1;
324  // sum weights
325  totalWeights += ((((uint16_t)1 << hfw0) >> 1) + (((uint16_t)1 << hfw1) >> 1));
326  }
327  remBytes = compressedSize - (1 + (hwCnt - 1) / 2) - 1;
328  // last weight calculation
329  huffDecoderTableLog = 1 + (31 - __builtin_clz(totalWeights));
330  // add last weight
331  uint16_t lw = (1 << huffDecoderTableLog) - totalWeights;
332  hufWeights[hwCnt++] = 1 + (31 - __builtin_clz(lw));
333  wCnt = hwCnt;
334  }
335  } else {
336  remBytes = compressedSize;
337  }
338  huffDecodeLiterals<PARALLEL_BYTE>(litDecodeInStream, quadStream, accHuff, bytesInAcc, remBytes,
339  regeneratedSize, huffDecoderTableLog, wCnt, hufWeights,
340  literalStream);
341  }
342  }
343  }
344 }

346 template <int PARALLEL_BYTE>
347 void parseBlockGenFSETable(hls::stream<bool>& blockValidStream,
348  hls::stream<uint32_t>& blockMetaStream,
349  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& zstdInStream,
350  hls::stream<bool>& litValidStream,
351  hls::stream<uint32_t>& litMetaStream,
352  hls::stream<uint32_t>& fseTableLitStream,
353  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& litDecodeInStream,
354  hls::stream<bool>& seqValidStream,
355  hls::stream<uint32_t>& seqMetaStream,
356  hls::stream<uint32_t>& fseTableSeqStream,
357  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& seqDecodeInStream) {
358  // parse blocks and decompress them
359  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
360  const uint16_t c_accRegWidth = c_streamWidth * 2;
361  const uint16_t c_accRegWidthx3 = c_streamWidth * 3;
362 
363  uint8_t defDistOffsets[3] = {0, c_maxCharLit + 1, c_maxCharLit + c_maxCharDefOffset + 2};
364  uint8_t maxCharLOM[3] = {c_maxCharLit, c_maxCharOffset, c_maxCharMatchlen};
365  int16_t prevDistribution[3 * 64];
366  uint8_t prevAccLog[3] = {6, 5, 6};
367  xfSymbolCompMode_t prevFseMode[3] = {FSE_COMPRESSED_MODE, FSE_COMPRESSED_MODE, FSE_COMPRESSED_MODE};
368 
369 parseBlock_main_loop:
370  while (blockValidStream.read()) {
371  // parse valid block and generate FSE tables for them
372  xfBlockType_t blockType;
373  uint32_t blockMeta = blockMetaStream.read(); // <blockType 8-bits><blockSize 24-bits>
374 
375  blockType = (xfBlockType_t)(blockMeta & 0x000000FF);
376  uint32_t blockSize = blockMeta >> 8;
377  // enable literal and decoding
378  litValidStream << 1;
379  seqValidStream << 1;
380 
381  // write common literal and sequence meta data
382  litMetaStream << blockMeta;
383  seqMetaStream << blockMeta;
384  // write literal data
385  if (blockType == RLE_BLOCK) {
386  // write RLE byte
387  ap_uint<c_streamWidth> tbuf = zstdInStream.read();
388  tbuf &= (ap_uint<c_streamWidth>)(((uint64_t)1 << 8) - 1);
389  litDecodeInStream << tbuf;
390  } else if (blockType == RAW_BLOCK) {
391  // write complete block data as it is
392  block_write_rawblocklit_data:
393  for (uint32_t i = 0; i < blockSize; i += PARALLEL_BYTE) {
394 #pragma HLS PIPELINE II = 1
395  ap_uint<c_streamWidth> tbuf = zstdInStream.read();
396  litDecodeInStream << tbuf;
397  }
398  } else {
399  // decompress zstd compressed blocks
400  /*
401  * Zstd compressed block contains Literals and Sequences sections
402  *
403  * Literals Section:
404  * <Literals_Section_Header><Huffman_Tree_Description><Jump_Table><Stream1><Stream2><Stream3><Stream4>
405  * Sequences Section:
406  * <Sequences_Section_Header><Literals_Length_Table><Offset_Table><Match_Length_Table><bitStream>
407  */
408  uint32_t regeneratedSize = 0; // literal count
409  uint32_t compressedSize = 0;
410  bool quadStream;
411  uint32_t remBlockSize = blockSize; // block header is not included in block size
412 
413  /*
414  * Literal_Section_Header
415  * <Literals_Block_Type><Size_Format><Regenerated_Size><Compressed_Size>
416  * 2 bits 1-2 bits 5-20 bits 0-18 bits
417  */
418  // read a word
419  ap_uint<c_accRegWidth> accRegister = zstdInStream.read();
420  uint8_t bytesInAcc = PARALLEL_BYTE;
421  uint8_t bitsInAcc = c_streamWidth;
422 
423  // Get Literals_Block_Type
424  xfLitBlockType_t litBlockType = (xfLitBlockType_t)((uint8_t)accRegister & 3);
425  accRegister >>= 2;
426  bitsInAcc -= 2;
427 
428  // Read size format
429  uint8_t sizeFormat = ((uint8_t)accRegister & 3);
430  uint8_t regSizeBits = 0;
431  bool getCompSize;
432  uint8_t ebn = 0; // extra bytes needed
433  // read regenerated and compressed size
434  getCompSize = (litBlockType == CMP_LBLOCK || litBlockType == TREELESS_LBLOCK);
435  if (getCompSize) {
436  accRegister >>= 2;
437  bitsInAcc -= 2;
438  ebn = 2;
439  switch (sizeFormat) {
440  case 0:
441  regSizeBits = 10;
442  quadStream = false;
443  break;
444  case 1:
445  regSizeBits = 10;
446  quadStream = true;
447  break;
448  case 2:
449  regSizeBits = 14;
450  quadStream = true;
451  ++ebn;
452  break;
453  case 3:
454  regSizeBits = 18;
455  quadStream = true;
456  ebn += 2;
457  break;
458  }
459  } else {
460  // get bits used by Regenerated_Size
461  if ((sizeFormat & 1) == 0) {
462  regSizeBits = 5;
463  accRegister >>= 1;
464  bitsInAcc -= 1;
465  } else {
466  accRegister >>= 2;
467  bitsInAcc -= 2;
468  if (sizeFormat == 1) {
469  regSizeBits = 12;
470  ++ebn;
471  } else { // remaining option is 3
472  regSizeBits = 20;
473  ebn += 2;
474  }
475  }
476  quadStream = false;
477  }
478 
479  // Read extra bytes
480  if (bitsInAcc < ((1 + ebn) * 8)) { // if less than required bits
481  accRegister.range((bitsInAcc + c_streamWidth - 1), bitsInAcc) = zstdInStream.read();
482  bitsInAcc += c_streamWidth;
483  }
484 
485  // read regenerated size
486  regeneratedSize = (accRegister & (((uint32_t)1 << regSizeBits) - 1));
487  accRegister >>= regSizeBits;
488  bitsInAcc -= regSizeBits;
489  // read compressed size
490  if (getCompSize) {
491  compressedSize = (accRegister & (((uint32_t)1 << regSizeBits) - 1));
492  accRegister >>= regSizeBits;
493  bitsInAcc -= regSizeBits;
494  }
495  // will be byte aligned compulsorily
496  bytesInAcc = bitsInAcc >> 3;
497  remBlockSize -= (ebn + 1);
498 
499  // Literal metadata format
500  // Word 0 is already written, which is default
501  // Word 1 <litBlockType><quadStream><regeneratedSize>
502  // 7-bits 1-bit 3 bytes
503  // Word 2 <huffmanHeader><compressedSize>
504  // 1 byte 3 bytes
505  // FSETables if present
506  uint32_t lmbuf = (uint8_t)litBlockType;
507  if (quadStream) lmbuf += (1 << 7);
508  lmbuf += (regeneratedSize << 8);
509  litMetaStream << lmbuf; // write word 1
510 
511  // write literals
512  uint32_t litSize2Write = regeneratedSize;
513  bool noTree = false;
514  switch (litBlockType) {
515  case RLE_LBLOCK: {
516  uint8_t rleLit;
517  if (bytesInAcc == 0) {
518  accRegister.range((bitsInAcc + c_streamWidth - 1), bitsInAcc) = zstdInStream.read();
519  bytesInAcc += PARALLEL_BYTE;
520  }
521  rleLit = ((uint8_t)accRegister & 0XFF);
522  accRegister >>= 8;
523  --bytesInAcc;
524  ap_uint<c_streamWidth> rlbuf = rleLit;
525  litDecodeInStream << rlbuf;
526 
527  bitsInAcc = bytesInAcc * 8;
528  remBlockSize -= 1;
529  break;
530  }
531  case TREELESS_LBLOCK: {
532  noTree = true;
533  }
534  case CMP_LBLOCK: {
535  litSize2Write = compressedSize;
536  // compressed literal block, FSE decode + huffman tree generation
537  // read from input if needed
538  if (bytesInAcc == 0) {
539  accRegister = zstdInStream.read();
540  bytesInAcc += PARALLEL_BYTE;
541  }
542  uint8_t hufHeader = 0;
543  if (noTree) {
544  // Word 2 <huffmanHeader><compressedSize>
545  // 1 byte 3 bytes
546  litMetaStream << (compressedSize << 8);
547  } else {
548  hufHeader = accRegister;
549  accRegister >>= 8;
550  --bytesInAcc;
551  --litSize2Write;
552  --remBlockSize;
553  // Word 2 <huffmanHeader><compressedSize>
554  // 1 byte 3 bytes
555  litMetaStream << (compressedSize << 8) + hufHeader; // write word 2
556 
557  // check if FSE decoding is required
558  if (hufHeader < 128) {
559  // FSE table generation and decoding before huffman decoding
560  uint8_t litAccuracyLog = 6; // max is 6 bits, will be modified as per input
561  // create FSE tables
562  int ret = generateFSETable<PARALLEL_BYTE>(fseTableLitStream, zstdInStream, accRegister,
563  bytesInAcc, litAccuracyLog, c_maxCharHuffman,
564  FSE_COMPRESSED_MODE, FSE_COMPRESSED_MODE,
565  c_defaultDistribution, prevDistribution);
566  litSize2Write -= ret;
567  remBlockSize -= ret;
568  // send relevant metadata
569  // Word 3 <accuracyLog><byteUsedByFSE>
570  // 1 byte 1 byte
571  litMetaStream << litAccuracyLog + ((uint32_t)ret << 8); // write word 3
572  }
573  }
574  bitsInAcc = bytesInAcc * 8;
575  }
576  case RAW_LBLOCK: {
577  // Bigger register needed, because more bytes than PARALLEL_BYTE can be present in the accumulator
578  ap_uint<c_accRegWidthx3> wbuf = accRegister;
579  uint8_t bytesWritten = 0;
580  uint8_t updBInAcc = bytesInAcc;
581 
582  bitsInAcc = bytesInAcc * 8;
583  // write block data
584  cmplit_write_block:
585  for (int i = 0; i < litSize2Write; i += PARALLEL_BYTE) {
586 #pragma HLS PIPELINE II = 1
587  if (i < (const int)(litSize2Write - bytesInAcc)) {
588  wbuf.range((bitsInAcc + c_streamWidth - 1), bitsInAcc) = zstdInStream.read();
589  updBInAcc += PARALLEL_BYTE;
590  }
591  ap_uint<c_streamWidth> tmpV = wbuf;
592  litDecodeInStream << tmpV;
593 
594  if (i > (const int)(litSize2Write - PARALLEL_BYTE)) {
595  bytesWritten = litSize2Write - i;
596  break;
597  }
598  wbuf >>= c_streamWidth;
599  updBInAcc -= PARALLEL_BYTE;
600  }
601  accRegister = (wbuf >> (bytesWritten * 8));
602  bytesInAcc = updBInAcc - bytesWritten;
603  bitsInAcc = bytesInAcc * 8;
604  remBlockSize -= litSize2Write;
605  } break;
606  default:
607  // must not reach here
608  break;
609  }
610  // decode zstd sequence header
611  /*
612  * Sequences Section:
613  * <Sequences_Section_Header><Literals_Length_Table><Offset_Table><Match_Length_Table><bitStream>
614  *
615  * Sequences_Section_Header
616  * <Number_of_Sequences><Symbol_Compression_Modes>
617  * 1-3 bytes 1 byte
618  */
619  if (bytesInAcc < remBlockSize && bytesInAcc < 4) {
620  accRegister.range((bitsInAcc + c_streamWidth - 1), bitsInAcc) = zstdInStream.read();
621  bytesInAcc += PARALLEL_BYTE;
622  }
623  uint8_t byteCnt = 0;
624  uint32_t seqCnt = 0;
625  xfSymbolCompMode_t literalsLengthMode;
626  xfSymbolCompMode_t offsetsMode;
627  xfSymbolCompMode_t matchLengthsMode;
628  // read 1 byte and check number of sequences
629  uint8_t byte_0 = (uint8_t)(accRegister & 0x000000FF);
630  accRegister >>= 8;
631  --bytesInAcc;
632  ++byteCnt;
633 
634  if (byte_0 == 0) {
635  // there are no sequences, sequence section stops here.
636  // Decompressed content is defined as Literals Section content.
637  // The FSE tables used in Repeat_Mode are not updated.
638  remBlockSize -= byteCnt;
639  // write sequence meta data
640  // Word 0 <blockMeta> ***already written
641  // Word 1 <1> if further decoding needed
642  // Word 2 <literalCount>
643  seqMetaStream << 0; // Word 1
644  seqMetaStream << regeneratedSize; // Word 2
645  } else {
646  // write sequence meta data
647  // Word 0 <blockMeta> ***already written
648  // Word 1 <decode_flag> 1 if further decoding needed, else 0
649  // Word 2 <literalCount>
650  seqMetaStream << 1; // Word 1
651  seqMetaStream << regeneratedSize; // Word 2
652 
653  if (byte_0 > 0 && byte_0 < 128) {
654  // uses one byte
655  seqCnt = byte_0;
656  } else if (byte_0 > 127 && byte_0 < 255) {
657  // uses 2 bytes
658  ++byteCnt;
659  // calculate Number_of_Sequences = ((byte0-128) << 8) + byte1
660  seqCnt = ((((uint32_t)byte_0 - 128) << 8) + ((uint32_t)accRegister & 0xFF));
661  accRegister >>= 8;
662  --bytesInAcc;
663  } else {
664  // uses 3 bytes, dump first byte, need next 2 bytes
665  // calculate Number_of_Sequences = byte1 + (byte2<<8) + 0x7F00
666  seqCnt = (((uint32_t)accRegister & 0x0000FFFF) + 32512);
667  byteCnt += 2;
668  accRegister >>= 16;
669  bytesInAcc -= 2;
670  }
671  // Get symbol compression mode
672  byte_0 = (uint8_t)accRegister;
673  accRegister >>= 8;
674  --bytesInAcc;
675  bitsInAcc = bytesInAcc * 8;
676  ++byteCnt;
677 
678  matchLengthsMode = (xfSymbolCompMode_t)((byte_0 >> 2) & 3);
679  offsetsMode = (xfSymbolCompMode_t)((byte_0 >> 4) & 3);
680  literalsLengthMode = (xfSymbolCompMode_t)((byte_0 >> 6) & 3);
681 
682  remBlockSize -= byteCnt;
683 
684  // generate FSE tables
685  ap_uint<c_accRegWidth> fseAcc = accRegister;
686  xfSymbolCompMode_t modeLOM[3] = {literalsLengthMode, offsetsMode, matchLengthsMode};
687  uint8_t accuracyLog[3] = {6, 5, 6}; // for default distribution, overwritten for fse compressed ones
688 #pragma HLS ARRAY_PARTITION variable = modeLOM complete
689 #pragma HLS ARRAY_PARTITION variable = accuracyLog complete
690 
691  if (offsetsMode == PREDEFINED_MODE)
692  maxCharLOM[1] = c_maxCharDefOffset;
693  else if (offsetsMode == FSE_COMPRESSED_MODE)
694  maxCharLOM[1] = c_maxCharOffset;
695 
696  gen_lom_fse_tables:
697  for (uint8_t i = 0; i < 3; ++i) {
698  // Generated FSE tables for literal lengths, offsets and match lengths
699  if (modeLOM[i] == REPEAT_MODE) accuracyLog[i] = prevAccLog[i];
700  int ret = generateFSETable<PARALLEL_BYTE>(
701  fseTableSeqStream, zstdInStream, fseAcc, bytesInAcc, accuracyLog[i], maxCharLOM[i], modeLOM[i],
702  prevFseMode[i], &(c_defaultDistribution[defDistOffsets[i]]), &(prevDistribution[i * 64]));
703  prevFseMode[i] = modeLOM[i];
704  prevAccLog[i] = accuracyLog[i];
705  remBlockSize -= ret;
706  }
707  accRegister = fseAcc;
708  bitsInAcc = bytesInAcc * 8;
709  // write sequence meta data
710  // Word 0 <blockMeta> ***already written
711  // Word 1 <decode_flag> 1 if further decoding needed, else 0
712  // Word 2 <literalCount>
713  // Word 3 <symbolCompressionMode><remBlockSize>
714  // 1 byte 3 bytes
715  // Word 4 <seqCnt>
716  // Word 5 <AccuracyLogs> --> <litlen><offset><matlen> 1 byte each lower-higher
717  uint32_t sqmbuf = (remBlockSize << 8) + byte_0;
718  seqMetaStream << sqmbuf; // Word 3
719  seqMetaStream << seqCnt; // Word 4
720  sqmbuf = (((uint32_t)accuracyLog[2] << 16) + ((uint32_t)accuracyLog[1] << 8) + accuracyLog[0]);
721  seqMetaStream << sqmbuf; // Word 5
722  // send the remaining block data, sequence bitstream
723  ap_uint<c_accRegWidthx3> wbuf = accRegister;
724  uint8_t bytesWritten = 0;
725  uint8_t updBInAcc = bytesInAcc;
726 
727  bitsInAcc = bytesInAcc * 8;
728  // write block data
729  cmpseq_write_block:
730  for (int i = 0; i < remBlockSize; i += PARALLEL_BYTE) {
731 #pragma HLS PIPELINE II = 1
732  if (i < (const int)(remBlockSize - bytesInAcc)) {
733  wbuf.range((bitsInAcc + c_streamWidth - 1), bitsInAcc) = zstdInStream.read();
734  updBInAcc += PARALLEL_BYTE;
735  }
736  ap_uint<c_streamWidth> tmpV = wbuf;
737  seqDecodeInStream << tmpV;
738 
739  if (i > (const int)(remBlockSize - PARALLEL_BYTE)) {
740  bytesWritten = remBlockSize - i;
741  break;
742  }
743  wbuf >>= c_streamWidth;
744  updBInAcc -= PARALLEL_BYTE;
745  }
746  }
747  }
748  }
749  litValidStream << 0;
750  seqValidStream << 0;
751 }

753 template <int PARALLEL_BYTE>
754 void parseFramesAndBlocks(hls::stream<ap_uint<8 * PARALLEL_BYTE> >& inStream,
755  hls::stream<ap_uint<4> >& inStrobe,
756  hls::stream<bool>& blockValidStream,
757  hls::stream<uint32_t>& blockMetaStream,
758  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& zstdBlockStream) {
759  // decompress all zstd frames
760  const uint8_t c_parallelBit = 8 * PARALLEL_BYTE;
761 
762  // 14 because of magic number + frame header
763  const uint8_t numWidth = 14 / PARALLEL_BYTE + 1;
764  ap_uint<128> inputWindow;
765  bool lastInputWord = false;
766  uint8_t inputIdx = 0;
767  ap_uint<4> istb;
768 // parse all frames and exit when there is no data to read in inStream
769 parseFrames_main_loop:
770  while (!lastInputWord) {
771  uint32_t magicNumber;
772  uint64_t windowSize = 0;
773  uint32_t dictionaryId = 0;
774  uint64_t frameContentSize = 0;
775  uint32_t contentCheckSum = 0;
776  uint32_t blockMaxSize = 0;
777  uint32_t readBytes = 0;
778  uint32_t processedBytes = 0;
779 
780  for (uint8_t i = 0; i < numWidth; i++) {
781 #pragma HLS PIPELINE II = 1
782  istb = inStrobe.read();
783  if (istb == 0) { // 0 data here means data ends here and there is no more frames to decode
784  lastInputWord = true;
785  ap_uint<c_parallelBit> dummyVal = inStream.read();
786  break;
787  } else {
788  inputWindow.range((i + 1) * c_parallelBit - 1, i * c_parallelBit) = inStream.read();
789  readBytes += PARALLEL_BYTE;
790  }
791  }
792 
793  if (lastInputWord) { // 0 data here means data ends here and there is no more frames to decode
794  break;
795  }
796 
797  /* Frame format
798  * <Magic_Number><Frame_Header><Data_Block(s).....><Content_Checksum>
799  * 4 bytes 2-14 bytes n bytes.... 0-4 bytes
800  */
801  // printf("Bytes in Acc: %d\n", bytesInAcc);
802  // Read data to accumulator
803  magicNumber = inputWindow.range(31, 0);
804  inputIdx += 4;
805 
806  // verify magic number
807  if (magicNumber != c_magicNumber) {
808  if ((magicNumber & c_skippableFrameMask) != c_skipFrameMagicNumber) {
809  // Error: Invalid Frame, magic number mismatch !!
810  return;
811  } else {
812  uint32_t frameSize = inputWindow >> inputIdx * 8;
813  inputIdx += 4;
814  processedBytes += 4;
815 
816  if ((readBytes - processedBytes) > frameSize) {
817  inputIdx += frameSize;
818  } else {
819  uint32_t skfRemBytes = frameSize - (readBytes - processedBytes);
820  uint32_t iterLim = 1 + ((skfRemBytes - 1) / PARALLEL_BYTE);
821  // skip this frame
822  ap_uint<c_parallelBit> input;
823  skip_frame_loop:
824  for (uint32_t i = 0; i < iterLim; ++i) {
825 #pragma HLS PIPELINE II = 1
826  istb = inStrobe.read();
827  input = inStream.read();
828  }
829  uint32_t remBytes = (PARALLEL_BYTE * iterLim) - skfRemBytes;
830  uint32_t nextFrameData = (PARALLEL_BYTE - remBytes);
831  input >>= remBytes * 8;
832  inputWindow.range(nextFrameData * 8 - 1, 0) = input;
833  inputIdx = nextFrameData;
834  }
835  }
836  continue;
837  }
838  /* Frame_Header format
839  * <Frame_Header_Descriptor><Window_Descriptor><Dictionary_Id><Frame_Content_Size>
840  * 1 byte 0-1 byte 0-4 bytes
841  * 0-8
842  * bytes
843  */
844  // Read data to accumulator
845  ap_uint<8> headerDesc = inputWindow.range(39, 32);
846  inputIdx += 1;
847 
848  // Parse frame header descriptor
849  /*
850  * Frame_Header_Descriptor
851  * <Frame_Content_Size_flag><Single_Segment_flag><Not used><Content_Checksum_flag><Dictionary_ID_flag>
852  * bit 7-6 bit 5 bit 4-3 bit 2 bit 1-0
853  */
854  bool checksumFlag = headerDesc.range(2, 2);
855  bool singleSegmentFlag = headerDesc.range(5, 5);
856  uint8_t didFieldSize = headerDesc.range(1, 0);
857  uint8_t fcsFieldSize = headerDesc.range(7, 6);
858 
859  didFieldSize += (uint8_t)(didFieldSize == 3);
860  fcsFieldSize = fcsFieldSize ? ((uint8_t)1 << fcsFieldSize) : (fcsFieldSize | singleSegmentFlag);
861  // Parse window descriptor
862  /*
863  * Calculate minimum buffer memory size (Window_Size) required for decompression for this frame.
864  * Minimum window size = 1KB and maximum Window_Size is ( (1 << 41) + 7 * (1 << 38) ) bytes, which is 3.75 TB.
865  *
866  * <Exponent><Mantissa>
867  * bit 7-3 bits 2-0
868  */
869  if (!singleSegmentFlag) {
870  // read 1 byte
871  uint8_t winDesc = (uint8_t)(inputWindow.range(47, 40));
872  inputIdx += 1;
873  // get window size
874  uint8_t windowLog = (10 + (winDesc >> 3));
875  uint64_t windowBase = (uint64_t)1 << windowLog;
876  uint64_t windowAdd = (windowBase >> 3) * (winDesc & 7);
877  windowSize = windowBase + windowAdd;
878  }
879 
880  // read Dictionary_Id
881  // read bytes if needed
882  dictionaryId = (didFieldSize > 0) ? inputWindow.range((inputIdx + didFieldSize) * 8 - 1, inputIdx * 8) : 0;
883  inputIdx += didFieldSize;
884 
885  // read Frame_Content_Size
886  frameContentSize = (fcsFieldSize > 0) ? inputWindow.range((inputIdx + fcsFieldSize) * 8 - 1, inputIdx * 8) : 0;
887  frameContentSize += ((fcsFieldSize == 2) ? 256 : 0);
888  inputIdx += fcsFieldSize;
889 
890  if (singleSegmentFlag) {
891  windowSize = frameContentSize;
892  }
893  // Block_Max_Size is minimum of window size and 128K
894  if (windowSize < (128 * 1024)) {
895  blockMaxSize = windowSize;
896  } else {
897  blockMaxSize = 128 * 1024;
898  }
899 
900  // parse Blocks in this Frame and pass block data to block decompression unit
901  bool isLastBlock = false;
902 
903  inputWindow >>= (inputIdx * 8);
904  uint8_t validBytes = readBytes - inputIdx;
905  processedBytes += inputIdx;
906  inputIdx = 0;
907 
908  parse_block_in_frame:
909  while (!isLastBlock) {
910  if (validBytes < 4) {
911  istb = inStrobe.read();
912  ap_uint<c_parallelBit> input = inStream.read();
913  inputWindow.range((validBytes + PARALLEL_BYTE) * 8 - 1, validBytes * 8) = input;
914  readBytes += PARALLEL_BYTE;
915  validBytes += PARALLEL_BYTE;
916  }
917 
918  // Parse block header
919  /*
920  * Block_Header: 3 bytes
921  * <Last_Block><Block_Type><Block_Size>
922  * bit 0 bits 1-2 bits 3-23
923  */
924  ap_uint<24> blockHeader = inputWindow.range(23, 0);
925  processedBytes += 3;
926 
927  inputWindow >>= 24;
928  validBytes -= 3;
929 
930  isLastBlock = (blockHeader & 1) ? true : false;
931  xfBlockType_t blockType = (xfBlockType_t)(((uint8_t)blockHeader >> 1) & 3);
932  uint32_t blockSize = (blockHeader >> 3);
933 
934  // enable processing for this block
935  blockValidStream << 1;
936 
937  // write metadata to blockMetaStream
938  uint32_t metabuf = (blockSize << 8) + ((uint8_t)blockType);
939  blockMetaStream << metabuf;
940  // copy blockSize data
941  uint32_t bs2Write = blockSize;
942 
943  if (blockType == RLE_BLOCK) {
944  bs2Write = 1;
945  }
946 
947  uint32_t blockLen = bs2Write;
948  int readCheck = bs2Write - validBytes;
949  uint8_t updBInAcc = validBytes;
950  uint8_t bytesWritten = 0;
951 
952  parser_write_block_data:
953  for (int i = 0; i < bs2Write; i += PARALLEL_BYTE) {
954 #pragma HLS PIPELINE II = 1
955  if (i < readCheck) {
956  istb = inStrobe.read();
957  inputWindow.range((validBytes + PARALLEL_BYTE) * 8 - 1, validBytes * 8) = inStream.read();
958  updBInAcc += PARALLEL_BYTE;
959  }
960 
961  ap_uint<c_parallelBit> outValue = inputWindow;
962  zstdBlockStream << outValue;
963 
964  if (i > (const int)(bs2Write - PARALLEL_BYTE)) {
965  bytesWritten = bs2Write - i;
966  break;
967  }
968 
969  inputWindow >>= c_parallelBit;
970  updBInAcc -= PARALLEL_BYTE;
971  }
972  validBytes = updBInAcc - bytesWritten;
973  inputWindow >>= bytesWritten * 8;
974  }
975 
976  // check for checksum flag
977  if (checksumFlag) {
978  if (validBytes < 4) {
979  istb = inStrobe.read();
980  ap_uint<c_parallelBit> input = inStream.read();
981  }
982  }
983  inputIdx = 0;
984  }
985  blockValidStream << 0;
986 }
