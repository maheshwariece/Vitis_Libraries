103  write_out_file_data:

109  }

112  tOut.keep = -1;

123  tOut.last = last;

134 template <int IN_DWIDTH>
135 void axiu2hlsStream(hls::stream<ap_axiu<IN_DWIDTH, 0, 0, 0> >& inAxiStream,
136  hls::stream<IntVectorStream_dt<8, IN_DWIDTH / 8> >& outHlsStream,
137  hls::stream<bool>& fileEos) {
138  constexpr int incr = IN_DWIDTH / 8;
139 // Iterate over all input files
140 fileAxi:
141  while (1) {
142  fileEos << false;
143  IntVectorStream_dt<8, incr> val;
144  ap_uint<IN_DWIDTH / 8> cntr = 0;
145  // Read AXI Stream
146  ap_axiu<IN_DWIDTH, 0, 0, 0> inData;
147 #ifdef __SYNTHESIS__
148  while (1) {
149  if (inAxiStream.read_nb(inData)) break;
150  }
151 #else
152  inData = inAxiStream.read();
153 #endif
154  bool last = inData.last;
155  ap_uint<IN_DWIDTH> tmpVal = inData.data;
156  ap_uint<IN_DWIDTH / 8> keep = inData.keep;
157  val.strobe = incr;
158 
159  {
160  axi2Hls:
161  while (!last) {
162 #pragma HLS PIPELINE
163  // Write the data in an aggregated manner in internal hls stream
164  for (auto i = 0, j = 0; i < IN_DWIDTH; i += incr) {
165 #pragma HLS UNROLL
166  val.data[j++] = tmpVal.range(i + 7, i);
167  }
168  outHlsStream << val;
169 
170  inData = inAxiStream.read();
171  last = inData.last;
172  tmpVal = inData.data;
173  keep = inData.keep;
174  }
175 
176  // Write the data in an aggregated manner in internal hls stream
177  for (auto i = 0, j = 0; i < IN_DWIDTH; i += incr) {
178 #pragma HLS UNROLL
179  val.data[j++] = tmpVal.range(i + 7, i);
180  }
181  while (keep) {
182  cntr += (keep & 0x1);
183  keep >>= 1;
184  }
185  val.strobe = cntr;
186  outHlsStream << val;
187  val.strobe = 0;
188  outHlsStream << val;
189  }
190 
191 // Check if any file is remaining
192 #ifndef __SYNTHESIS__
193  if (inData.last) break;
194 #endif
195  }
196  fileEos << true;
197 }

408 template <uint16_t STREAMDWIDTH, class SIZE_DT = uint32_t, uint16_t AXIWIDTH = 32>
409 void streamDm2k(hls::stream<ap_uint<STREAMDWIDTH> >& in,
410  hls::stream<SIZE_DT>& inSize,
411  hls::stream<ap_axiu<STREAMDWIDTH, 0, 0, 0> >& outStream) {
423  // read data from input hls to input stream for decompression kernel
424 
425  constexpr int c_size = STREAMDWIDTH / 8;
426  SIZE_DT inputSize = inSize.read();
427  SIZE_DT inputAlignedSize = inputSize / c_size;
428  uint8_t inputModSize = inputSize % c_size;
429 alignedTransfer:
430  for (auto i = 0; i < inputAlignedSize; i++) {
431 #pragma HLS PIPELINE II = 1
432  ap_uint<STREAMDWIDTH> v = in.read();
433  ap_axiu<STREAMDWIDTH, 0, 0, 0> inData;
434  inData.data = v;
435  inData.keep = -1;
436  inData.last = false;
437  if (inputModSize == 0 && i == inputAlignedSize - 1) {
438  inData.last = true;
439  }
440  outStream << inData;
441  }
442 
443  if (inputModSize) {
444  ap_uint<STREAMDWIDTH> v = in.read();
445  ap_axiu<STREAMDWIDTH, 0, 0, 0> inData;
446  inData.data = v;
447  inData.last = true;
448  uint32_t num = 0;
449  keepCalc:
450  for (auto b = 0; b < inputModSize; b++) {
451  num |= 1UL << b;
452  }
453  inData.keep = num;
454  outStream << inData;
455  }
