76 template <int MAX_FREQ_DWIDTH = 18, int MIN_MATCH = 3>
77 void zstdLz77DivideStream(hls::stream<IntVectorStream_dt<32, 1> >& inStream,
78  hls::stream<IntVectorStream_dt<8, 1> >& litStream,
79  hls::stream<DSVectorStream_dt<Sequence_dt<MAX_FREQ_DWIDTH>, 1> >& seqStream,
80  hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& litFreqStream,
81  hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& seqFreqStream,
82  hls::stream<bool>& rleFlagStream,
83  hls::stream<ap_uint<MAX_FREQ_DWIDTH> >& metaStream) {
84  // lz77 encoder states
85  enum LZ77EncoderStates { WRITE_LITERAL, WRITE_OFFSET0, WRITE_OFFSET1 };
86  // offsets in frequency table for literals->litlen->matlen->offset
87  // constexpr uint16_t offsets = {0, c_maxLitV + 1, c_maxLitV + c_maxCodeLL + 2, c_maxLitV + c_maxCodeLL +
88  // c_maxCodeML + 3};
89  IntVectorStream_dt<8, 1> outLitVal;
90  DSVectorStream_dt<Sequence_dt<MAX_FREQ_DWIDTH>, 1> outSeqVal;
91  IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> outLitFreqVal;
92  IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> outSeqFreqVal;
93  ap_uint<MAX_FREQ_DWIDTH> metaVal = 0;
94  bool last_block = false;
95  bool just_started = true;
96  int blk_n = 0;
97 
98  while (!last_block) {
99  // iterate over multiple blocks in a file
100  ++blk_n;
101  enum LZ77EncoderStates next_state = WRITE_LITERAL;
102  ap_uint<MAX_FREQ_DWIDTH> literal_freq[c_maxLitV + 1];
103  ap_uint<MAX_FREQ_DWIDTH> litlen_freq[c_maxCodeLL + 1];
104  ap_uint<MAX_FREQ_DWIDTH> matlen_freq[c_maxCodeML + 1];
105  ap_uint<MAX_FREQ_DWIDTH> offset_freq[c_maxCodeOF + 1];
106 
107  ap_uint<MAX_FREQ_DWIDTH> seqCnt = 0;
108  {
109 #pragma HLS LOOP_MERGE force
110  lit_freq_init:
111  for (uint16_t i = 0; i < c_maxLitV + 1; i++) {
112  literal_freq[i] = 0;
113  }
114  ll_freq_init:
115  for (uint16_t i = 0; i < c_maxCodeLL + 1; i++) {
116  litlen_freq[i] = 0;
117  }
118  ml_freq_init:
119  for (uint16_t i = 0; i < c_maxCodeML + 1; i++) {
120  matlen_freq[i] = 0;
121  }
122  of_freq_init:
123  for (uint16_t i = 0; i < c_maxCodeOF + 1; i++) {
124  offset_freq[i] = 0;
125  }
126  }
127  uint8_t tCh = 0;
128  uint8_t tLen = 0;
129  ap_uint<MAX_FREQ_DWIDTH> litCount = 0;
130  ap_uint<MAX_FREQ_DWIDTH> litTotal = 0;
131  // set output data to be of valid length
132  outLitVal.strobe = 1;
133  outSeqVal.strobe = 1;
134  uint8_t cLit = 0;
135  bool fv = true;
136  bool isRLE = true;
137  zstd_lz77_divide:
138  while (true) {
139 #pragma HLS PIPELINE II = 1
140 #ifndef DISABLE_DEPENDENCE
141 #pragma HLS dependence variable = literal_freq inter false
142 #pragma HLS dependence variable = litlen_freq inter false
143 #pragma HLS dependence variable = matlen_freq inter false
144 #pragma HLS dependence variable = offset_freq inter false
145 #endif
146  // read value from stream
147  auto encodedValue = inStream.read();
148  if (encodedValue.strobe == 0) {
149  last_block = just_started;
150  just_started = true;
151  break;
152  }
153  just_started = false;
154  tCh = encodedValue.data[0].range(7, 0);
155  tLen = encodedValue.data[0].range(15, 8);
156  uint16_t tOffset = encodedValue.data[0].range(31, 16) + 3 + 1;
157 
158  if (tLen) {
159  // if match length present, get sequence codes
160  uint8_t llc = getLLCode(litCount);
161  uint8_t mlc = getMLCode(tLen - MIN_MATCH);
162  uint8_t ofc = bitsUsed31(tOffset);
163  // reset code frequencies
164  litlen_freq[llc]++;
165  matlen_freq[mlc]++;
166  offset_freq[ofc]++;
167  // write sequence
168  outSeqVal.data[0].litlen = litCount;
169  outSeqVal.data[0].matlen = tLen - MIN_MATCH;
170  outSeqVal.data[0].offset = tOffset;
171  seqStream << outSeqVal;
172  // if (blk_n == 2) printf("ll: %d, ml: %d, of: %d\n", litCount, tLen - MIN_MATCH, tOffset);
173  litCount = 0;
174  ++seqCnt;
175  } else {
176  // store first literal to check for RLE block
177  if (fv) {
178  cLit = tCh;
179  } else {
180  if (cLit != tCh) isRLE = false;
181  }
182  fv = false;
183  // increment literal count
184  literal_freq[tCh]++;
185  ++litCount;
186  ++litTotal;
187  // write literal
188  outLitVal.data[0] = tCh;
189  litStream << outLitVal;
190  // printf("%c", (char)tCh);
191  }
192  }
193  if (!last_block) {
194  isRLE = (isRLE && cLit == 0);
195  if (isRLE) {
196  // printf("RLE Case blk_n: %u\n", blk_n);
197  literal_freq[3]++;
198  outLitVal.data[0] = 3;
199  litStream << outLitVal;
200  }
201  rleFlagStream << isRLE;
202  }
203  outLitVal.strobe = 0;
204  outSeqVal.strobe = 0;
205  litStream << outLitVal;
206  seqStream << outSeqVal;
207  // write literal and distance trees
208  if (!last_block) {
209  metaStream << (ap_uint<MAX_FREQ_DWIDTH>)litTotal;
210  metaStream << (ap_uint<MAX_FREQ_DWIDTH>)seqCnt;
211  outLitFreqVal.strobe = 1;
212  outSeqFreqVal.strobe = 1;
213  // if (blk_n == 1) printf("Literals freq\n");
214  write_lit_freq:
215  for (ap_uint<9> i = 0; i < c_maxLitV + 1; ++i) {
216 #pragma HLS PIPELINE II = 1
217  outLitFreqVal.data[0] = literal_freq[i];
218  litFreqStream << outLitFreqVal;
219  // if (blk_n == 1) printf("%d. freq: %u\n", (uint16_t)i, (uint16_t)literal_freq[i]);
220  }
221  // first write total number of sequences
222  // printf("seqCnt: %u\n", (uint32_t)seqCnt);
223  outSeqFreqVal.data[0] = seqCnt;
224  seqFreqStream << outSeqFreqVal;
225  // write last ll, ml, of codes
226  outSeqFreqVal.data[0] = getLLCode(outSeqVal.data[0].litlen);
227  seqFreqStream << outSeqFreqVal;
228  outSeqFreqVal.data[0] = getMLCode(outSeqVal.data[0].matlen);
229  seqFreqStream << outSeqFreqVal;
230  outSeqFreqVal.data[0] = bitsUsed31(outSeqVal.data[0].offset);
231  seqFreqStream << outSeqFreqVal;
232  // if (blk_n == 1) printf("Litlen freq\n");
233  write_lln_freq:
234  for (ap_uint<9> i = 0; i < c_maxCodeLL + 1; ++i) {
235 #pragma HLS PIPELINE II = 1
236  outSeqFreqVal.data[0] = litlen_freq[i];
237  seqFreqStream << outSeqFreqVal;
238  // if (blk_n == 1) printf("%d. freq: %u\n", (uint16_t)i, (uint16_t)litlen_freq[i]);
239  }
240  // if (blk_n == 12) printf("Offset freq\n");
241  write_ofs_freq:
242  for (ap_uint<9> i = 0; i < c_maxCodeOF + 1; ++i) {
243 #pragma HLS PIPELINE II = 1
244  outSeqFreqVal.data[0] = offset_freq[i];
245  seqFreqStream << outSeqFreqVal;
246  // if (blk_n == 12) printf("%d. freq: %u\n", (uint16_t)i, (uint16_t)offset_freq[i]);
247  }
248  // if (blk_n == 12) printf("Matlen freq\n");
249  write_mln_freq:
250  for (ap_uint<9> i = 0; i < c_maxCodeML + 1; ++i) {
251 #pragma HLS PIPELINE II = 1
252  outSeqFreqVal.data[0] = matlen_freq[i];
253  seqFreqStream << outSeqFreqVal;
254  // if (blk_n == 12) printf("%d. freq: %u\n", (uint16_t)i, (uint16_t)matlen_freq[i]);
255  }
256  }
257  }
258  // eos needed only to indicated end of block
259  outLitFreqVal.strobe = 0;
260  outSeqFreqVal.strobe = 0;
261  litFreqStream << outLitFreqVal;
262  seqFreqStream << outSeqFreqVal;
263  metaStream << (ap_uint<MAX_FREQ_DWIDTH>)(-1);
264 }

284 void getLitSequences(hls::stream<IntVectorStream_dt<8, 1> >& inStream,
285  hls::stream<IntVectorStream_dt<8, 1> >& litStream,
286  hls::stream<DSVectorStream_dt<Sequence_dt<MAX_FREQ_DWIDTH>, 1> >& seqStream,
287  hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& litFreqStream,
288  hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& seqFreqStream,
289  hls::stream<bool>& rleFlagStream,
290  hls::stream<ap_uint<MAX_FREQ_DWIDTH> >& outMetaStream) {
291 #pragma HLS dataflow
292  hls::stream<IntVectorStream_dt<32, 1> > compressedStream("compressedStream");
293  hls::stream<IntVectorStream_dt<32, 1> > boosterStream("boosterStream");
294  hls::stream<ap_uint<MAX_FREQ_DWIDTH> > metaStream("metaStream");
295 #pragma HLS STREAM variable = compressedStream depth = 16
296 #pragma HLS STREAM variable = boosterStream depth = 16
297 #pragma HLS STREAM variable = metaStream depth = 8
298 
299  // LZ77 compress
300  xf::compression::lzCompress<MAX_BLOCK_SIZE, uint32_t, MATCH_LEN, MIN_MATCH, LZ_MAX_OFFSET_LIMIT>(inStream,
301  compressedStream);
302  // improve CR and generate clean sequences
303  xf::compression::lzBooster<MAX_MATCH_LEN>(compressedStream, boosterStream);
304  // separate literals from sequences and generate literal frequencies
305  xf::compression::details::zstdLz77DivideStream<MAX_FREQ_DWIDTH, MIN_MATCH>(
306  boosterStream, litStream, seqStream, litFreqStream, seqFreqStream, rleFlagStream, metaStream);
307  xf::compression::details::zstdCompressionMeta<MAX_FREQ_DWIDTH>(metaStream, outMetaStream);
308 }

310 template <int BLOCK_SIZE = 32 * 1024, int MAX_FREQ_DWIDTH = 17, int PARALLEL_LIT_STREAMS = 4>
311 void preProcessLitStream(hls::stream<IntVectorStream_dt<8, 1> >& inLitStream,
312  hls::stream<IntVectorStream_dt<8, 1> >& outReversedLitStream) {
313  /*
314  * Description: This module reads literals from input stream and divides them into 1 or 4 streams
315  * of equal size. Last stream can be upto 3 bytes less than other 3 streams. This module
316  * streams literals in reverse order of input.
317  */
318  // constexpr int c_litStrMaxSize = 1 + ((BLOCK_SIZE - 1) / PARALLEL_LIT_STREAMS);
319  // literal buffer
320  ap_uint<8> litBuffer[BLOCK_SIZE];
321  IntVectorStream_dt<8, 1> outLit;
322 
323  while (true) {
324  uint16_t litIdx = 0;
325  uint16_t litCnt = 0;
326  uint8_t streamCnt = 1;
327  uint16_t streamSize[4] = {0, 0, 0, 0};
328  // write to buffer and read from buffer
329  read_literals_bf:
330  for (auto inVal = inLitStream.read(); inVal.strobe > 0; inVal = inLitStream.read()) {
331 #pragma HLS PIPELINE II = 1
332  litBuffer[litIdx++] = inVal.data[0];
333  ++litCnt;
334  }
335  if (litCnt == 0) break;
336  // get out stream count
337  if (litCnt > 255) {
338  streamCnt = 4;
339  streamSize[0] = 1 + (litCnt - 1) / 4;
340  streamSize[1] = streamSize[0];
341  streamSize[2] = streamSize[0];
342  streamSize[3] = litCnt - (streamSize[0] * 3);
343  } else {
344  streamSize[0] = litCnt;
345  }
346  outLit.strobe = 1;
347  // first send the stream count
348  outLit.data[0] = streamCnt;
349  outReversedLitStream << outLit;
350 
351  int16_t idxL = -1;
352  int16_t idxH = -1;
353  int16_t litIdxRange = 0;
354  write_rev_literals_outer:
355  for (uint8_t k = 0; k < streamCnt; ++k) {
356  idxL = idxH;
357  idxH += streamSize[k];
358  // First write sub-stream size, followed by the stream data
359  outLit.data[0] = (uint8_t)(streamSize[k]);
360  outReversedLitStream << outLit;
361  outLit.data[0] = (uint8_t)(streamSize[k] >> 8);
362  outReversedLitStream << outLit;
363  // printf("LitStream %d, size: %d\nStream Reversed\n", k, streamSize[k]);
364 
365  write_rev_literals:
366  for (int16_t idx = idxH; idx > idxL; --idx) {
367 #pragma HLS PIPELINE II = 1
368  // write to output huffman encoder unit in reverse
369  outLit.data[0] = litBuffer[idx];
370  outReversedLitStream << outLit;
371  // printf("%d, ", (uint8_t)litBuffer[idx]);
372  }
373  // printf("\n");
374  }
375  // end of block indicator not needed, since size is also sent
376  // outLit.strobe = 0;
377  // outReversedLitStream << outLit;
378  }
379  // end of data
380  outLit.strobe = 0;
381  outReversedLitStream << outLit;
382 }

384 template <int BLOCK_SIZE = 32 * 1024, int MAX_FREQ_DWIDTH = 17, int MIN_MATCH = 3>
385 void reverseSeq(hls::stream<DSVectorStream_dt<Sequence_dt<MAX_FREQ_DWIDTH>, 1> >& seqStream,
386  hls::stream<DSVectorStream_dt<Sequence_dt<MAX_FREQ_DWIDTH>, 1> >& outReversedSeqStream,

440 template <int MAX_FREQ_DWIDTH = 17>
441 void frequencySequencer(hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& wghtFreqStream,
442  hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& seqFreqStream,
443  hls::stream<IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> >& freqStream) {
444  // sequence input frequencies into single output stream
445  constexpr uint16_t c_limits[4] = {c_maxCodeLL + 1, c_maxCodeOF + 1, c_maxZstdHfBits + 1, c_maxCodeML + 1};
446  const uint8_t c_hfIdx = 2;
447  IntVectorStream_dt<MAX_FREQ_DWIDTH, 1> outfreq;
448  while (true) {
449  outfreq = seqFreqStream.read();
450  if (outfreq.strobe == 0) break; // will come only at the end of all data
451  // first value is sequences count
452  freqStream << outfreq;
453  // next three values are last ll, ml, of codes
454  write_last_seq:
455  for (uint8_t i = 0; i < 3; ++i) {
456 #pragma HLS PIPELINE II = 1
457  outfreq = seqFreqStream.read();
458  freqStream << outfreq;
459  }
460  write_bl_ll_ml_of:
461  for (uint8_t fIdx = 0; fIdx < 4; ++fIdx) {
462  auto llim = c_limits[fIdx];
463  bool isBlen = (c_hfIdx == fIdx);
464  if (isBlen) {
465  // maxVal from literals
466  outfreq = wghtFreqStream.read();
467  freqStream << outfreq;
468  }
469  // first send the size, followed by data
470  outfreq.data[0] = llim;
471  freqStream << outfreq;
472  write_inp_freq:
473  for (uint16_t i = 0; i < llim; ++i) {
474 #pragma HLS PIPELINE II = 1
475  if (isBlen) {
476  outfreq = wghtFreqStream.read();
477  } else {
478  outfreq = seqFreqStream.read();
479  }
480  freqStream << outfreq;
481  }
482  // dump strobe 0
483  if (isBlen) wghtFreqStream.read();
484  }
485  }
486  // dump strobe 0
487  wghtFreqStream.read();
488  outfreq.data[0] = 0;
489  outfreq.strobe = 0;
490  freqStream << outfreq;
491 }
