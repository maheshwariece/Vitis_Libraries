61  processedBytes += PARALLEL_BYTES;
62

66  processedBytes += PARALLEL_BYTES;
67  }
68

70 #pragma HLS PIPELINE II = 1

73  }
74

76  dt_blockInfo blockInfo;
77  uint8_t incrInputIdx = 0;
78 
79  while ((processedBytes + inputIdx) < inputSize) {
80  if (inputIdx >= PARALLEL_BYTES) {
81  inputWindow >>= c_parallelBit;
82  inputIdx = inputIdx - PARALLEL_BYTES;
83  processedBytes += PARALLEL_BYTES;
84  if (readBytes < inputSize) {
85  ap_uint<c_parallelBit> input = inStream.read();
86  inputWindow.range(8 * c_parallelBit - 1, 7 * c_parallelBit) = input;
87  readBytes += PARALLEL_BYTES;
88  }
89  }
90 
91  SIZE_DT chunkSize = 0;
92 
93  ap_uint<32> inValue = inputWindow >> (inputIdx * 8);
94  chunkSize = inValue.range(31, 8);
95 
96  origCompLen = chunkSize - 4;
97  compLen = origCompLen;
98 
99  // 4 bytes processed and remaining 4 bytes skipped
100  incrInputIdx = 8;
101 
102  uint8_t chunkIdx = inValue.range(7, 0);
103 
104  if (chunkIdx == 0x01) {
105  blockInfo.storedBlock = 1;
106  } else {
107  blockInfo.storedBlock = 0;
108  uint8_t blkSizeProcessedBytes = 0;
109  ap_uint<16> value = inputWindow >> ((inputIdx + incrInputIdx) * 8);
110 
111  bool c0 = ((value.range(7, 7)) == 1);
112  bool c1 = ((value.range(15, 15)) == 1);
113  if (c0 & c1) {
114  incrInputIdx = 11;
115  blkSizeProcessedBytes = 3;
116  } else if (c0) {
117  incrInputIdx = 10;
118  blkSizeProcessedBytes = 2;
119  } else {
120  incrInputIdx = 9;
121  blkSizeProcessedBytes = 1;
122  }
123  origCompLen = origCompLen - blkSizeProcessedBytes;
124  compLen = origCompLen;
125  }
126 
127  inputIdx += incrInputIdx;
128  blockInfo.compressSize = origCompLen;
129  // write blockInfo to stream
130  blockInfoStream << blockInfo;
131 
132  SIZE_DT len = compLen;
133  for (uint32_t blockData = 0; blockData < compLen; blockData += PARALLEL_BYTES) {
134 #pragma HLS PIPELINE II = 1
135  if (inputIdx >= PARALLEL_BYTES) {
136  inputWindow >>= c_parallelBit;
137  inputIdx = inputIdx - PARALLEL_BYTES;
138  processedBytes += PARALLEL_BYTES;
139  if (readBytes < inputSize) {
140  ap_uint<c_parallelBit> input = inStream.read();
141  inputWindow.range(8 * c_parallelBit - 1, 7 * c_parallelBit) = input;
142  readBytes += PARALLEL_BYTES;
143  }
144  }
145 
146  outStreamValue = inputWindow >> (inputIdx * 8);
147  outStream << outStreamValue;
148 
149  if (len >= PARALLEL_BYTES) {
150  inputIdx += PARALLEL_BYTES;
151  len -= PARALLEL_BYTES;
152  } else {
153  inputIdx += len;
154  len = 0;
155  }
156  }
157  }
158  blockInfo.compressSize = 0;
159  // writing 0 to indicate end of data
160  blockInfoStream << blockInfo;
161 }

298 template <int NUM_BLOCKS, int PARALLEL_BYTES, int BLOCK_SIZE = 64, class SIZE_DT = ap_uint<17> >
299 void lzMultiBlockPacker(hls::stream<ap_uint<(PARALLEL_BYTES * 8) + 8> > lzDataStream[NUM_BLOCKS],
300  hls::stream<uint32_t> lzSizeStream[NUM_BLOCKS],
301  hls::stream<ap_uint<(PARALLEL_BYTES * 8) + 8> >& outStream,
302  hls::stream<uint32_t>& outSizeStream) {
303  uint32_t outSize = 0;
304  const uint8_t c_parallelBit = PARALLEL_BYTES * 8;
305  const uint8_t c_streamWidth = (PARALLEL_BYTES * 8) + 8;
306  uint32_t blockSizeInBytes = BLOCK_SIZE * 1024;
307  ap_uint<NUM_BLOCKS> is_pending;
308  bool done = false;
309 
310  for (uint8_t i = 0; i < NUM_BLOCKS; i++) {
311 #pragma HLS UNROLL
312  is_pending.range(i, i) = 1;
313  }
314 
315  while (is_pending) {
316  for (int i = 0; i < NUM_BLOCKS; i++) {
317  for (SIZE_DT read = 0; ((read < blockSizeInBytes) && (is_pending.range(i, i) == true));
318  read += PARALLEL_BYTES) {
319 #pragma HLS PIPELINE II = 1
320  ap_uint<c_streamWidth> outData = lzDataStream[i].read();
321  bool eosFlag = outData.range(c_streamWidth - 1, c_parallelBit);
322  if (eosFlag == false) {
323  outStream << outData;
324  } else {
325  is_pending.range(i, i) = 0;
326  }
327  }
328  }
329  }
330 
331  ap_uint<c_streamWidth> outData = 0;
332  outData.range(c_streamWidth - 1, c_parallelBit) = 1;
333  outStream << outData;
334 
335  for (int i = 0; i < NUM_BLOCKS; i++) {
336 #pragma HLS PIPELINE II = 1
337  outSize += lzSizeStream[i].read();
338  }
339  outSizeStream << outSize;
340 }
