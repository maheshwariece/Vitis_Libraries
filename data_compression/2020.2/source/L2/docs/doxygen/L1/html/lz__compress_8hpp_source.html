339 void lzCompress(hls::stream<IntVectorStream_dt<8, 1> >& inStream, hls::stream<IntVectorStream_dt<32, 1> >& outStream) {
340  const uint16_t c_indxBitCnts = 24;
341  const uint16_t c_fifo_depth = LEFT_BYTES + 2;
342  const int c_dictEleWidth = (MATCH_LEN * 8 + c_indxBitCnts);
343  typedef ap_uint<MATCH_LEVEL * c_dictEleWidth> uintDictV_t;
344  typedef ap_uint<c_dictEleWidth> uintDict_t;
345  const uint32_t totalDictSize = (1 << (c_indxBitCnts - 1)); // 8MB based on index 3 bytes
346  uint32_t relativeInSize = 0;
347  bool resetDictFlag = true;
348  uint32_t relativeNumBlocks = 0;
349 
350  uintDictV_t dict[LZ_DICT_SIZE];
351 #pragma HLS RESOURCE variable = dict core = XPM_MEMORY uram
352 
353  // local buffers for each block
354  uint8_t present_window[MATCH_LEN];
355 #pragma HLS ARRAY_PARTITION variable = present_window complete
356  hls::stream<uint8_t> lclBufStream("lclBufStream");
357 #pragma HLS STREAM variable = lclBufStream depth = c_fifo_depth
358 #pragma HLS BIND_STORAGE variable = lclBufStream type = fifo impl = srl
359 
360  // input register
361  IntVectorStream_dt<8, 1> inVal;
362  // output register
363  IntVectorStream_dt<32, 1> outValue;
364  // loop over blocks
365  while (true) {
366  uint32_t iIdx = 0;
367  // once 8MB data is processed reset dictionary
368  // 8MB based on index 3 bytes
369  if (resetDictFlag) {
370  ap_uint<MATCH_LEVEL* c_dictEleWidth> resetValue = 0;

381  }
382  resetDictFlag = false;
383  relativeInSize = 0;
384  relativeNumBlocks = 0;
385  } else {
386  relativeNumBlocks++;
387  }
388  // check if end of data
389  auto nextVal = inStream.read();
390  if (nextVal.strobe == 0) {
391  outValue.strobe = 0;
392  outStream << outValue;
393  break;
394  }
395  // fill buffer and present_window
396  lz_fill_present_win:
397  while (iIdx < MATCH_LEN - 1) {
398 #pragma HLS PIPELINE II = 1
399  inVal = nextVal;
400  nextVal = inStream.read();
401  present_window[++iIdx] = inVal.data[0];
402  ++relativeInSize;
403  }
404  // assuming that, at least bytes more than LEFT_BYTES will be present at the input
405  lz_fill_circular_buf:
406  for (uint16_t i = 0; i < LEFT_BYTES; ++i) {
407 #pragma HLS PIPELINE II = 1
408  inVal = nextVal;
409  nextVal = inStream.read();
410  lclBufStream << inVal.data[0];
411  ++relativeInSize;
412  }
413  // lz_compress main
414  outValue.strobe = 1;
415 
416  lz_compress:
417  for (; nextVal.strobe != 0; ++iIdx) {
418 #pragma HLS PIPELINE II = 1
419 #ifndef DISABLE_DEPENDENCE
420 #pragma HLS dependence variable = dict inter false
421 #endif
422  uint32_t currIdx = (iIdx + (relativeNumBlocks * MAX_INPUT_SIZE)) - MATCH_LEN + 1;
423  // read from input stream into circular buffer
424  auto inValue = lclBufStream.read(); // pop latest value from FIFO
425  lclBufStream << nextVal.data[0]; // push latest read value to FIFO
426  nextVal = inStream.read(); // read next value from input stream
427  ++relativeInSize;
428 
429  // shift present window and load next value
430  for (uint8_t m = 0; m < MATCH_LEN - 1; m++) {
431 #pragma HLS UNROLL
432  present_window[m] = present_window[m + 1];
433  }
434 
435  present_window[MATCH_LEN - 1] = inValue;
436 
437  // Calculate Hash Value
438  uint32_t hash =
439  (present_window[0] << 4) ^ (present_window[1] << 3) ^ (present_window[2] << 2) ^ (present_window[3]);
440 
441  // Dictionary Lookup
442  uintDictV_t dictReadValue = dict[hash];
443  uintDictV_t dictWriteValue = dictReadValue << c_dictEleWidth;
444  for (int m = 0; m < MATCH_LEN; m++) {
445 #pragma HLS UNROLL

451

455  uint32_t match_offset = 0;
456  for (int l = 0; l < MATCH_LEVEL; l++) {
457  uint8_t len = 0;
