154  hls::stream<uint32_t> coreSizeStream("coreSizeStream");
155  hls::stream<ap_uint<c_dwidth> > distStream[c_numBlocks];

159  hls::stream<ap_uint<17> > upsizedCntr[c_numBlocks];

169  hls::stream<ap_uint<c_dwidth + c_size_dwidth> > mergeStream[c_numBlocks];

171 #pragma HLS STREAM variable = checksumStream depth = 4
172 #pragma HLS STREAM variable = checksumSizeStream depth = 4
173 #pragma HLS STREAM variable = checksumEos depth = 4
174 #pragma HLS STREAM variable = coreStream depth = 4
175 #pragma HLS STREAM variable = coreSizeStream depth = 4
176 #pragma HLS STREAM variable = distStream depth = c_uramFifoDepth
177 #pragma HLS STREAM variable = strdSizeStream depth = 4

205  // send input data to both checksum and for compression
206  xf::compression::details::dataDuplicator<c_dwidth, c_checksumParallelBytes>(
207  inStream, inSizeStream, checksumStream, checksumEos, checksumSizeStream, coreStream, coreSizeStream);

209  // checksum kernel
210  xf::compression::checksum32<c_checksumParallelBytes>(checksumInitStream, checksumStream, checksumSizeStream,
211  checksumEos, checksumOutStream, checksumOutEos,
212  checksumTypeStream);
213 
214  // distrubute block size data into each block in round-robin fashion
215  details::multicoreDistributor<c_freq_dwidth, c_dwidth, c_numBlocks, c_blockSize>(
216  coreStream, coreSizeStream, strdStream, strdSizeStream, distStream, distSizeStream);
217 
218  // Parallel LZ77
219  for (uint8_t i = 0; i < c_numBlocks; i++) {
220 #pragma HLS UNROLL
221  xf::compression::details::streamDownSizerSize<c_dwidth, 8, c_freq_dwidth>(distStream[i], distSizeStream[i],
222  downStream[i]);
223 
224  lz77Compress<c_blockSize, c_freq_dwidth>(downStream[i], lz77Stream[i], lz77Tree[i]);
225 
226  if (URAM_BUFFERS == 1) {
227  xf::compression::details::bufferUpsizer<9, 72>(lz77Stream[i], lz77UpsizedStream[i], upsizedCntr[i]);
228  } else {
229  xf::compression::details::sendBuffer<9>(lz77Stream[i], lz77PassStream[i], upsizedCntr[i]);
230  }
231  }
232 
233  // Single Call Treegen
234  details::zlibTreegenStreamWrapper<c_numBlocks>(lz77Tree, hufCodeStream);
235 
236  // Parallel Huffman
237  for (uint8_t i = 0; i < c_numBlocks; i++) {
238 #pragma HLS UNROLL
239  if (URAM_BUFFERS == 1) {
240  xf::compression::details::bufferDownsizer<72, 9>(lz77UpsizedStream[i], lz77DownsizedStream[i],
241  upsizedCntr[i]);
242  } else {
243  xf::compression::details::receiveBuffer<9>(lz77PassStream[i], lz77DownsizedStream[i], upsizedCntr[i]);
244  }
245 
246  zlibHuffmanEncoder(lz77DownsizedStream[i], hufCodeStream[i], huffStream[i]);
247 
248  xf::compression::details::simpleStreamUpsizer<16, c_dwidth, c_size_dwidth>(huffStream[i], mergeStream[i]);
249  }
250 
251  // read all num block data in round-robin fashion and write into single outstream
252  details::multicoreMerger<c_dwidth, c_size_dwidth, c_numBlocks, c_blockSize>(mergeStream, strdStream, strdSizeStream,
253  outStream, outStreamEos, outSizeStream);
254 }
