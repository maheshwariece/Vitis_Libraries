49 template <int MAX_LIT_COUNT, int PARALLEL_UNITS>
50 static void lz4CompressPart1(hls::stream<ap_uint<32> >& inStream,
51  hls::stream<uint8_t>& lit_outStream,
52  hls::stream<ap_uint<64> >& lenOffset_Stream,
53  uint32_t input_size,
54  uint32_t max_lit_limit[PARALLEL_UNITS],
55  uint32_t index) {
56  if (input_size == 0) return;
57 
58  uint8_t match_len = 0;
59  uint32_t lit_count = 0;
60  uint32_t lit_count_flag = 0;
61 
62  ap_uint<32> nextEncodedValue = inStream.read();
63 lz4_divide:
64  for (uint32_t i = 0; i < input_size;) {
65 #pragma HLS PIPELINE II = 1
66  ap_uint<32> tmpEncodedValue = nextEncodedValue;
67  if (i < (input_size - 1)) nextEncodedValue = inStream.read();
68  uint8_t tCh = tmpEncodedValue.range(7, 0);
69  uint8_t tLen = tmpEncodedValue.range(15, 8);
70  uint16_t tOffset = tmpEncodedValue.range(31, 16);
71  uint32_t match_offset = tOffset;
72 
73  if (lit_count >= MAX_LIT_COUNT) {
74  lit_count_flag = 1;
75  } else if (tLen) {
76  uint8_t match_len = tLen - 4; // LZ4 standard
77  ap_uint<64> tmpValue;
78  tmpValue.range(63, 32) = lit_count;
79  tmpValue.range(15, 0) = match_len;
80  tmpValue.range(31, 16) = match_offset;
81  lenOffset_Stream << tmpValue;
82  match_len = tLen - 1;
83  lit_count = 0;
84  } else {
85  lit_outStream << tCh;
86  lit_count++;
87  }
88  if (tLen)
89  i += tLen;
90  else
91  i += 1;
92  }
93  if (lit_count) {
94  ap_uint<64> tmpValue;
95  tmpValue.range(63, 32) = lit_count;
96  if (lit_count == MAX_LIT_COUNT) {
97  lit_count_flag = 1;
98  tmpValue.range(15, 0) = 777;
99  tmpValue.range(31, 16) = 777;
100  } else {
101  tmpValue.range(15, 0) = 0;
102  tmpValue.range(31, 16) = 0;
103  }
104  lenOffset_Stream << tmpValue;
105  }
106  max_lit_limit[index] = lit_count_flag;
107 }

109 static void lz4CompressPart2(hls::stream<uint8_t>& in_lit_inStream,
110  hls::stream<ap_uint<64> >& in_lenOffset_Stream,
111  hls::stream<ap_uint<8> >& outStream,
112  hls::stream<bool>& endOfStream,
113  hls::stream<uint32_t>& compressdSizeStream,
114  uint32_t input_size) {
115  // LZ4 Compress STATES
116  enum lz4CompressStates { WRITE_TOKEN, WRITE_LIT_LEN, WRITE_MATCH_LEN, WRITE_LITERAL, WRITE_OFFSET0, WRITE_OFFSET1 };
117  uint32_t lit_len = 0;
118  uint16_t outCntr = 0;
119  uint32_t compressedSize = 0;
120  enum lz4CompressStates next_state = WRITE_TOKEN;
121  uint16_t lit_length = 0;
122  uint16_t match_length = 0;
123  uint16_t write_lit_length = 0;
124  ap_uint<16> match_offset = 0;
125  bool lit_ending = false;
126  bool extra_match_len = false;
127  bool readOffsetFlag = true;
128 
129 lz4_compress:
130  for (uint32_t inIdx = 0; (inIdx < input_size) || (!readOffsetFlag);) {
131 #pragma HLS PIPELINE II = 1
132  ap_uint<8> outValue;
133  ap_uint<64> nextLenOffsetValue;
134 
135  if (readOffsetFlag) {
136  nextLenOffsetValue = in_lenOffset_Stream.read();
137  readOffsetFlag = false;
138  }
139 
140  if (next_state == WRITE_TOKEN) {
141  lit_length = nextLenOffsetValue.range(63, 32);
142  match_length = nextLenOffsetValue.range(15, 0);
143  match_offset = nextLenOffsetValue.range(31, 16);
144  inIdx += match_length + lit_length + 4;
145 
146  if (match_length == 777 && match_offset == 777) {
147  inIdx = input_size;
148  lit_ending = true;
149  }
150 
151  lit_len = lit_length;
152  write_lit_length = lit_length;
153  if (match_offset == 0 && match_length == 0) {
154  lit_ending = true;
155  }
156  if (lit_length >= 15) {
157  outValue.range(7, 4) = 15;
158  lit_length -= 15;
159  next_state = WRITE_LIT_LEN;
160  readOffsetFlag = false;
161  } else if (lit_length) {
162  outValue.range(7, 4) = lit_length;
163  lit_length = 0;
164  next_state = WRITE_LITERAL;
165  readOffsetFlag = false;
166  } else {
167  outValue.range(7, 4) = 0;
168  next_state = WRITE_OFFSET0;
169  readOffsetFlag = false;
170  }
171  if (match_length >= 15) {
172  outValue.range(3, 0) = 15;
173  match_length -= 15;
174  extra_match_len = true;
175  } else {
176  outValue.range(3, 0) = match_length;
177  match_length = 0;
178  extra_match_len = false;
179  }
180  } else if (next_state == WRITE_LIT_LEN) {
181  if (lit_length >= 255) {
182  outValue = 255;
183  lit_length -= 255;
184  } else {
185  outValue = lit_length;
186  next_state = WRITE_LITERAL;
187  readOffsetFlag = false;
188  }
189  } else if (next_state == WRITE_LITERAL) {
190  outValue = in_lit_inStream.read();
191  write_lit_length--;
192  if (write_lit_length == 0) {
193  if (lit_ending) {
194  next_state = WRITE_TOKEN;
195  readOffsetFlag = true;
196  } else {
197  next_state = WRITE_OFFSET0;
198  readOffsetFlag = false;
199  }
200  }
201  } else if (next_state == WRITE_OFFSET0) {
202  match_offset++; // LZ4 standard
203  outValue = match_offset.range(7, 0);
204  next_state = WRITE_OFFSET1;
205  readOffsetFlag = false;
206  } else if (next_state == WRITE_OFFSET1) {
207  outValue = match_offset.range(15, 8);
208  if (extra_match_len) {
209  next_state = WRITE_MATCH_LEN;
210  readOffsetFlag = false;
211  } else {
212  next_state = WRITE_TOKEN;
213  readOffsetFlag = true;
214  }
215  } else if (next_state == WRITE_MATCH_LEN) {
216  if (match_length >= 255) {
217  outValue = 255;
218  match_length -= 255;
219  } else {
220  outValue = match_length;
221  next_state = WRITE_TOKEN;
222  readOffsetFlag = true;
223  }
224  }
225  if (compressedSize < input_size) {
226  // Limiting compression size not more than input size.
227  // Host code should ignore such blocks
228  outStream << outValue;
229  endOfStream << 0;
230  compressedSize++;
231  }
232  }
233 
234  compressdSizeStream << compressedSize;
235  outStream << 0;
236  endOfStream << 1;
237 }

363  int DATAWIDTH = 512,
364  int BURST_SIZE = 16,
365  int NUM_BLOCK = 8,

372  int MIN_B_SIZE = 128>
373 void lz4CompressMM(const data_t* in, data_t* out, uint32_t* compressd_size, const uint32_t input_size) {
374  uint32_t block_idx = 0;
375  uint32_t block_length = 64 * 1024;
376  uint32_t no_blocks = (input_size - 1) / block_length + 1;
377  uint32_t max_block_size = 64 * 1024;
378  uint32_t readBlockSize = 0;
379 
380  bool small_block[NUM_BLOCK];
381  uint32_t input_block_size[NUM_BLOCK];
382  uint32_t input_idx[NUM_BLOCK];
383  uint32_t output_idx[NUM_BLOCK];
384  uint32_t output_block_size[NUM_BLOCK];
385  uint32_t max_lit_limit[NUM_BLOCK];
386  uint32_t small_block_inSize[NUM_BLOCK];
387 #pragma HLS ARRAY_PARTITION variable = input_block_size dim = 0 complete
388 #pragma HLS ARRAY_PARTITION variable = input_idx dim = 0 complete
389 #pragma HLS ARRAY_PARTITION variable = output_idx dim = 0 complete
390 #pragma HLS ARRAY_PARTITION variable = output_block_size dim = 0 complete
391 #pragma HLS ARRAY_PARTITION variable = max_lit_limit dim = 0 complete
392 
393  // Figure out total blocks & block sizes
394  for (uint32_t i = 0; i < no_blocks; i += NUM_BLOCK) {
395  uint32_t nblocks = NUM_BLOCK;
396  if ((i + NUM_BLOCK) > no_blocks) {
397  nblocks = no_blocks - i;
398  }
399 
400  for (uint32_t j = 0; j < NUM_BLOCK; j++) {
401  if (j < nblocks) {
402  uint32_t inBlockSize = block_length;
403  if (readBlockSize + block_length > input_size) inBlockSize = input_size - readBlockSize;
404  if (inBlockSize < MIN_B_SIZE) {
405  small_block[j] = 1;
406  small_block_inSize[j] = inBlockSize;
407  input_block_size[j] = 0;
408  input_idx[j] = 0;
409  } else {
410  small_block[j] = 0;
411  input_block_size[j] = inBlockSize;
412  readBlockSize += inBlockSize;
413  input_idx[j] = (i + j) * max_block_size;
414  output_idx[j] = (i + j) * max_block_size;
415  }
416  } else {
417  input_block_size[j] = 0;
418  input_idx[j] = 0;
419  }
420  output_block_size[j] = 0;
421  max_lit_limit[j] = 0;
422  }
423 
424  // Call for parallel compression
425  hlsLz4<data_t, DATAWIDTH, BURST_SIZE, NUM_BLOCK>(in, out, input_idx, output_idx, input_block_size,
426  output_block_size, max_lit_limit);
427 
428  for (uint32_t k = 0; k < nblocks; k++) {
429  if (max_lit_limit[k]) {
430  compressd_size[block_idx] = input_block_size[k];
431  } else {
432  compressd_size[block_idx] = output_block_size[k];
433  }
434 
435  if (small_block[k] == 1) {
436  compressd_size[block_idx] = small_block_inSize[k];
437  }
438  block_idx++;
439  }
440  }
441 }
