71 template <int DWIDTH = 64, int SIZE_DWIDTH = 4, int PARALLEL_BYTES = 8, int STRATEGY = 0>
72 void dataDuplicator(hls::stream<IntVectorStream_dt<8, PARALLEL_BYTES> >& inStream,
73  hls::stream<ap_uint<32> >& checksumInitStream,
74  hls::stream<ap_uint<PARALLEL_BYTES * 8> >& checksumOutStream,
75  hls::stream<ap_uint<5> >& checksumSizeStream,
76  hls::stream<bool>& checksumOutEos,
77  hls::stream<bool>& fileEos,
78  hls::stream<ap_uint<DWIDTH + SIZE_DWIDTH> >& coreStream) {
79  constexpr int incr = DWIDTH / 8;
80 
81  for (bool eosFile = fileEos.read(); eosFile == false; eosFile = fileEos.read()) {
82  // Checksum initial data
83  if (STRATEGY == 0) {
84  checksumInitStream << ~0;
85  } else {
86  checksumInitStream << 0;
87  }
88 
89  checksumOutEos << false;
90 
91  duplicator:
92  while (1) {
93  IntVectorStream_dt<8, PARALLEL_BYTES> tmpVal = inStream.read();
94  // Last will be high if strobe is 0
95  bool last = (tmpVal.strobe == 0);
96  ap_uint<DWIDTH + SIZE_DWIDTH> outVal = 0;
97  // First SIZE_DWIDTH bits will be no. of valid bytes
98  outVal.range(SIZE_DWIDTH - 1, 0) = tmpVal.strobe;
99  // Checksum requires the parallel valid bytes
100  checksumSizeStream << (ap_uint<5>)tmpVal.strobe;
101  // Data parallel write
102  for (auto i = SIZE_DWIDTH, j = 0; i < (DWIDTH + SIZE_DWIDTH); i += incr) {
103 #pragma HLS UNROLL
104  outVal.range(i + 7, i) = tmpVal.data[j++];
105  }
106  // Core Data Stream
107  coreStream << outVal;
108  if (last) break;
109  // Checksum Data Stream
110  checksumOutStream << outVal.range(DWIDTH + SIZE_DWIDTH - 1, SIZE_DWIDTH);
111  }
112  }
113  checksumOutEos << true;
114 }

145 template <int FREQ_DWIDTH = 32, int DWIDTH = 64, int NUM_BLOCKS = 4, int BLOCK_SIZE = 32768, int MIN_BLCK_SIZE = 64>
146 void multicoreDistributor(hls::stream<ap_uint<DWIDTH> >& inStream,
147  hls::stream<uint32_t>& inSizeStream,
148  hls::stream<ap_uint<DWIDTH> >& strdStream,
149  hls::stream<ap_uint<16> >& strdSizeStream,
150  hls::stream<ap_uint<DWIDTH> > distStream[NUM_BLOCKS],
151  hls::stream<ap_uint<FREQ_DWIDTH> > distSizeStream[NUM_BLOCKS]) {

153  ap_uint<4> core = 0;
154  uint32_t readSize = 0;
155  uint32_t inputSize = inSizeStream.read();
156

158  core %= NUM_BLOCKS;
159  ap_uint<FREQ_DWIDTH> outputSize = ((readSize + BLOCK_SIZE) > inputSize) ? (inputSize - readSize) : BLOCK_SIZE;
160 
161  bool storedBlockFlag = (outputSize <= MIN_BLCK_SIZE);
162 
163  readSize += outputSize;
164  if (!storedBlockFlag)
165  distSizeStream[core] << outputSize;
166  else

169  multicoreDistributor:

171 #pragma HLS PIPELINE II = 1
172  ap_uint<DWIDTH> outData = inStream.read();
173  if (storedBlockFlag)
174  strdStream << outData;
175  else
176  distStream[core] << outData;
177  }
178  core++;
179  }
180 
181  for (int i = 0; i < NUM_BLOCKS; i++) {
182  distSizeStream[i] << 0;
183  }
184  strdSizeStream << 0;
185 }

187 template <class SIZE_DT = uint64_t,
188  int DWIDTH = 64,
189  int SIZE_DWIDTH = 4,
190  int NUM_BLOCKS = 4,
191  int BLOCK_SIZE = 32768,
192  int MIN_BLCK_SIZE = 64,
193  int STRATEGY = 0>
194 void multicoreDistributor(hls::stream<ap_uint<DWIDTH + SIZE_DWIDTH> >& inStream,
195  hls::stream<SIZE_DT>& fileSizeStream,
196  hls::stream<ap_uint<DWIDTH> >& strdStream,
197  hls::stream<ap_uint<16> >& strdSizeStream,
198  hls::stream<bool>& blckEosStream,
199  hls::stream<bool>& fileEos,
200  hls::stream<ap_uint<DWIDTH + SIZE_DWIDTH> > distStream[NUM_BLOCKS]) {
201  constexpr int c_incr = DWIDTH / 8;
202  constexpr int c_factor = (MIN_BLCK_SIZE + c_incr) / c_incr;
203  constexpr int c_storedDepth = 2 * c_factor;
204  ap_uint<4> core = 0;
205 
206  while (1) {
207  if (fileEos.read()) {
208  break;
209  }
210  SIZE_DT readSize = 0;
211  SIZE_DT writeSize = 0;
212  ap_uint<16> strdCntr = 0;
213  ap_uint<SIZE_DWIDTH> strobe = 0;
214  bool last = false;
215 
216  hls::stream<ap_uint<DWIDTH + SIZE_DWIDTH> > storedBufferStream;
217 #pragma HLS STREAM variable = storedBufferStream depth = c_storedDepth
218 
219  init_loop:
220  for (uint8_t i = 0; i < c_factor && !last; i++) {
221 #pragma HLS PIPELINE II = 1
222  ap_uint<DWIDTH + SIZE_DWIDTH> tmpVal = inStream.read();
223  strobe = tmpVal.range(SIZE_DWIDTH - 1, 0);
224  last = (strobe == 0);
225  readSize += strobe;
226  storedBufferStream << tmpVal;
227  }
228 
229  distributor:
230  while (!last) {
231  core %= NUM_BLOCKS;
232  blckEosStream << false;
233 
234  for (uint32_t i = 0; i < BLOCK_SIZE && !last; i += c_incr) {
235 #pragma HLS PIPELINE II = 1
236  ap_uint<DWIDTH + SIZE_DWIDTH> tmpVal = inStream.read();
237  ap_uint<SIZE_DWIDTH> strobe = tmpVal.range(SIZE_DWIDTH - 1, 0);
238  readSize += strobe;
239  last = (strobe == 0);
240  ap_uint<DWIDTH + SIZE_DWIDTH> tmp = storedBufferStream.read();
241  distStream[core] << tmp;
242  strdCntr += tmp.range(SIZE_DWIDTH - 1, 0);
243  writeSize += tmp.range(SIZE_DWIDTH - 1, 0);
244  storedBufferStream << tmpVal;
245  }
246 
247  if (!last) {
248  distStream[core] << 0;
249  strdCntr = 0;
250  core++;
251  }
252  }
253 
254  blckEosStream << true;
255  bool onlyOnce = true;
256  bool endBlck = true;
257  for (; writeSize != readSize && strdCntr != 0;) {
258 #pragma HLS PIPELINE II = 1
259  ap_uint<SIZE_DWIDTH + DWIDTH> tmpVal = storedBufferStream.read();
260  distStream[core] << tmpVal;
261  strdCntr += tmpVal.range(SIZE_DWIDTH - 1, 0);
262  writeSize += tmpVal.range(SIZE_DWIDTH - 1, 0);
263  if (strdCntr == BLOCK_SIZE) {
264  endBlck = false;
265  strdCntr = 0;
266  }
267  }
268 
269  if (!endBlck) {
270  if (readSize == writeSize)
271  distStream[core] << storedBufferStream.read();
272  else
273  distStream[core] << 0;
274  core++;
275  } else if (endBlck && strdCntr > 0) {
276  distStream[core] << storedBufferStream.read();
277  core++;
278  }
279 
280  for (; (writeSize != readSize && strdCntr == 0) || onlyOnce;) {
281 #pragma HLS PIPELINE II = 1
282  if (onlyOnce) {
283  strdSizeStream << (readSize - writeSize);
284  onlyOnce = false;
285  if (readSize == writeSize) break;
286  }
287  ap_uint<SIZE_DWIDTH + DWIDTH> tmpVal = storedBufferStream.read();
288  strdStream << tmpVal.range(SIZE_DWIDTH + DWIDTH - 1, SIZE_DWIDTH);
289  writeSize += tmpVal.range(SIZE_DWIDTH - 1, 0);
290  if (readSize == writeSize) storedBufferStream.read();
291  }
292 
293  // Total Input Size for GZIP only
294  if (STRATEGY == 0) fileSizeStream << writeSize;
295  }
296 
297 ip_terminate_data:
298  for (uint8_t i = 0; i < NUM_BLOCKS; i++) {
299  distStream[i] << 0;
300  }
301 }

303 template <int DWIDTH = 64, int SIZE_DWIDTH = 4, int NUM_BLOCKS = 4, int BLOCK_SIZE = 32768, int MIN_BLCK_SIZE = 64>
304 void multicoreMerger(hls::stream<ap_uint<DWIDTH + SIZE_DWIDTH> > inStream[NUM_BLOCKS],
305  hls::stream<ap_uint<DWIDTH> >& strdStream,
306  hls::stream<ap_uint<16> >& strdSizeStream,
307  hls::stream<ap_uint<DWIDTH> >& outStream,
308  hls::stream<bool>& outStreamEos,
309  hls::stream<uint32_t>& outSizeStream) {
310  constexpr int incr = DWIDTH / 8;
311  uint32_t outSize = 0;
312  ap_uint<NUM_BLOCKS> is_pending;
313  ap_uint<2 * DWIDTH> inputWindow;
314  uint32_t factor = DWIDTH / 8;
315  uint32_t inputIdx = 0;
316 
317  for (uint8_t i = 0; i < NUM_BLOCKS; i++) {
318 #pragma HLS UNROLL
319  is_pending.range(i, i) = 1;
320  }
321 
322  while (is_pending) {
323  for (int i = 0; i < NUM_BLOCKS; i++) {
324  bool blockDone = false;
325  bool not_first = false;
326  for (; (blockDone == false) && (is_pending(i, i) == true);) {
327 #pragma HLS PIPELINE II = 1
328  assert((inputIdx + factor) <= 2 * (DWIDTH / 8));
329 
330  ap_uint<DWIDTH + SIZE_DWIDTH> inVal = inStream[i].read();
331  uint8_t inSize = inVal.range(SIZE_DWIDTH - 1, 0);
332  outSize += inSize;
333  inputWindow.range((inputIdx + factor) * 8 - 1, inputIdx * 8) =
334  inVal.range(SIZE_DWIDTH + DWIDTH - 1, SIZE_DWIDTH);
335 
336  ap_uint<DWIDTH> outData = inputWindow.range(DWIDTH - 1, 0);
337  inputIdx += inSize;
338 
339  // checks per engine end of data or not, use inSize to avoid eos
340  blockDone = (inSize == 0);
341  is_pending.range(i, i) = (not_first || inSize > 0);
342 
343  if (inputIdx >= factor) {
344  outStream << outData;
345  outStreamEos << 0;
346  inputWindow >>= DWIDTH;
347  inputIdx -= factor;
348  }
349  not_first = true;
350  }
351  }
352  }
353 
354  int16_t storedSize = strdSizeStream.read();
355  outSize += storedSize;
356  for (; storedSize > 0; storedSize -= factor) {
357  // adding stored block header
358  inputWindow.range((inputIdx + 1) * 8 - 1, inputIdx * 8) = 0;
359  inputIdx += 1;
360  inputWindow.range((inputIdx + 2) * 8 - 1, inputIdx * 8) = storedSize;
361  inputIdx += 2;
362  inputWindow.range((inputIdx + 2) * 8 - 1, inputIdx * 8) = ~storedSize;
363  inputIdx += 2;
364  outSize += 5;
365 
366  // process stored block data
367  inputWindow.range((inputIdx + factor) * 8 - 1, inputIdx * 8) = strdStream.read();
368  ap_uint<DWIDTH> outData = inputWindow.range(DWIDTH - 1, 0);
369  inputIdx += factor;
370  if (inputIdx >= factor) {
371  outStream << outData;
372  outStreamEos << 0;
373  inputWindow >>= DWIDTH;
374  inputIdx -= factor;
375  }
376  }
377 
378  if (outSize <= MIN_BLCK_SIZE) storedSize = strdSizeStream.read();
379 
380  if (inputIdx) {
381  outStream << inputWindow.range(DWIDTH - 1, 0);
382  outStreamEos << 0;
383  }
384 
385  // send end of stream data
386  outStream << 0;
387  outStreamEos << 1;
388  outSizeStream << outSize;
389 }

391 template <int NUM_BLOCKS = 8, int STRATEGY = 0>
392 void gzipZlibPackerEngine(hls::stream<ap_uint<68> > inStream[NUM_BLOCKS],
393  hls::stream<ap_uint<68> >& packedStream,
394  hls::stream<ap_uint<64> >& strdStream,
395  hls::stream<ap_uint<16> >& strdSizeStream,
396  hls::stream<uint32_t>& fileSizeStream,
397  hls::stream<ap_uint<32> >& checksumStream,
398  hls::stream<bool>& checksumStreamEos,
399  hls::stream<bool>& fileEos,
400  hls::stream<bool>& blckEosStream) {
401  ap_uint<68> tmpVal = 0;
402  ap_uint<4> core = 0;
403 
404  while (1) {
405  if (fileEos.read()) {
406  checksumStreamEos.read();
407  break;
408  }
409  // Header Handling
410  if (STRATEGY == 0) { // GZIP
411  tmpVal.range(67, 0) = 0x0000000008088B1F8;
412  packedStream << tmpVal;
413  tmpVal.range(67, 0) = 0x007803004;
414  packedStream << tmpVal;
415  } else { // ZLIB
416  tmpVal.range(67, 0) = 0x01782;
417  packedStream << tmpVal;
418  }
419 
420  // Compressed Data Handling
421  blckHandler:
422  for (bool blckEos = blckEosStream.read(); blckEos == false; blckEos = blckEosStream.read()) {
423  core %= NUM_BLOCKS;
424  bool blockDone = false;
425  for (; blockDone == false;) {
426 #pragma HLS PIPELINE II = 1
427  ap_uint<68> inVal = inStream[core].read();
428  blockDone = (inVal.range(3, 0) == 0);
429  if (blockDone) break;
430  packedStream << inVal;
431  }
432  core++;
433  }
434 
435  // Stored Block Header
436  ap_uint<16> sizeVal = strdSizeStream.read();
437  bool strdFlg = (sizeVal != 0);
438  if (strdFlg) {
439  tmpVal = ~sizeVal;
440  tmpVal <<= 16;
441  tmpVal.range(15, 0) = sizeVal;
442  tmpVal <<= 12;
443  tmpVal.range(11, 0) = 5;
444  packedStream << tmpVal;
445  }
446 
447  // Stored Block
448  strdBlck:
449  for (uint16_t size = 0; size < sizeVal; size += 8) {
450 #pragma HLS PIPELINE II = 1
451  uint8_t rSize = 8;
452  if (rSize + size > sizeVal) rSize = sizeVal - size;
453  tmpVal.range(3, 0) = rSize;
454  tmpVal.range(67, 4) = strdStream.read();
455  packedStream << tmpVal;
456  }
457 
458  // Checksum Data
459  while (1) {
460  bool eos = checksumStreamEos.read();
461  // Last Block Data
462  tmpVal.range(67, 0) = 0xffff0000015;
463  packedStream << tmpVal;
464 
465  tmpVal.range(67, 4) = checksumStream.read();
466  tmpVal.range(3, 0) = 4;
467  packedStream << tmpVal;
468  // Input Size Data
469  if (STRATEGY == 0) {
470  tmpVal.range(67, 4) = fileSizeStream.read();
471  tmpVal.range(3, 0) = 4;
472  packedStream << tmpVal;
473  }
474  tmpVal = 0;
475  packedStream << tmpVal;
476  if (!eos) break;
477  }
478  }
479  for (auto i = 0; i < NUM_BLOCKS; i++) inStream[i].read();
480 }
